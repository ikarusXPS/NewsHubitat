```typescript
/**
 * Build Cloudinary URL for responsive images
 * Falls back to original URL if Cloudinary not configured
 */
export function buildCloudinaryUrl(
  originalUrl: string,
  width: number,
  format: 'avif' | 'webp' | 'jpg' = 'jpg'
): string {
  // Cloudinary fetch URL pattern
  const cloudName = import.meta.env.VITE_CLOUDINARY_CLOUD_NAME;

  // Graceful fallback if no Cloudinary configured
  if (!cloudName) return originalUrl;

  // Cloudinary fetch mode: f_format,q_auto,w_width
  return `https://res.cloudinary.com/${cloudName}/image/fetch/f_${format},q_auto,w_${width}/${encodeURIComponent(originalUrl)}`;
}
```

**Key decisions:**
- `q_auto` = automatic quality optimization (Cloudinary chooses based on format)
- `encodeURIComponent` = prevent URL injection (per RESEARCH.md Security Domain)
- Fallback to original URL if `VITE_CLOUDINARY_CLOUD_NAME` not set (per RESEARCH.md A1)
- Format parameter supports AVIF/WebP/JPEG (D-75)

**Container structure:**
```tsx
<div ref={ref} className={cn('relative overflow-hidden', className)}>
  {/* Blur placeholder */}
  {!loaded && (
    <div className="absolute inset-0 bg-gradient-to-br from-[#0a0e1a] to-[#050810] animate-pulse" />
  )}

  {/* Conditional load based on priority or inView */}
  {shouldLoad && (
    <picture>
      {/* AVIF source */}
      <source
        type="image/avif"
        srcSet={widths.map(w => `${buildCloudinaryUrl(src, w, 'avif')} ${w}w`).join(', ')}
        sizes={sizes || "(max-width: 768px) 100vw, 50vw"}
      />

      {/* WebP source */}
      <source
        type="image/webp"
        srcSet={widths.map(w => `${buildCloudinaryUrl(src, w, 'webp')} ${w}w`).join(', ')}
        sizes={sizes || "(max-width: 768px) 100vw, 50vw"}
      />

      {/* JPEG fallback */}
      <img
        src={buildCloudinaryUrl(src, 960, 'jpg')}
        srcSet={widths.map(w => `${buildCloudinaryUrl(src, w, 'jpg')} ${w}w`).join(', ')}
        sizes={sizes || "(max-width: 768px) 100vw, 50vw"}
        alt={alt}
        loading={priority ? 'eager' : 'lazy'}
        decoding="async"
        fetchPriority={priority ? 'high' : 'auto'}
        onLoad={() => setLoaded(true)}
        onError={handleImageError}
        className={cn(
          'h-full w-full object-cover transition-opacity duration-300',
          loaded ? 'opacity-100' : 'opacity-0'
        )}
      />
    </picture>
  )}

  {/* Error state */}
  {imageError && (
    <div className="absolute inset-0 flex items-center justify-center bg-gray-900/50">
      <ImageOff className="h-8 w-8 text-gray-600" />
    </div>
  )}
</div>
```
