3. Create `prometheus/alert.rules.yml` (D-18, D-19, D-20):

```yaml
# NewsHub Alert Rules
# See: .planning/phases/20-monitoring-alerting/20-CONTEXT.md

groups:
  - name: newshub
    rules:
      # Error rate alert (D-18): 5xx/total > 1% over 5 minutes
      - alert: HighErrorRate
        expr: |
          (
            sum(rate(http_requests_total{status_code=~"5.."}[5m]))
            /
            sum(rate(http_requests_total[5m]))
          ) > 0.01
        for: 5m
        labels:
          severity: critical
        annotations:
          summary: "High error rate detected"
          description: "Error rate is {{ $value | humanizePercentage }} over the last 5 minutes"

      # Latency alert (D-19): p95 > 2 seconds over 5 minutes
      - alert: HighLatency
        expr: |
          histogram_quantile(0.95, sum(rate(http_request_duration_seconds_bucket[5m])) by (le))
          > 2
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "High p95 latency detected"
          description: "p95 latency is {{ $value | humanizeDuration }}"

      # Service down alert
      - alert: ServiceDown
        expr: up{service="newshub"} == 0
        for: 1m
        labels:
          severity: critical
        annotations:
          summary: "NewsHub service is down"
          description: "The NewsHub application has been down for more than 1 minute"
```

```yaml
# Alertmanager configuration for NewsHub
# See: .planning/phases/20-monitoring-alerting/20-CONTEXT.md
#
# Environment variables are substituted at container startup:
# - SMTP_HOST, SMTP_PORT, SMTP_USER, SMTP_PASS (from existing .env)
# - ALERT_EMAIL (recipient for alert notifications)

global:
  smtp_smarthost: '${SMTP_HOST}:${SMTP_PORT}'
  smtp_from: 'alertmanager@newshub.local'
  smtp_auth_username: '${SMTP_USER}'
  smtp_auth_password: '${SMTP_PASS}'
  smtp_require_tls: true

route:
  group_by: ['alertname']     # D-22: Group by alert name
  group_wait: 30s             # D-22: Wait 30s before sending grouped alert
  group_interval: 5m          # Time between grouped notifications
  repeat_interval: 4h         # Don't repeat same alert for 4 hours
  receiver: 'email-notifications'

receivers:
  - name: 'email-notifications'   # D-23: Email receiver
    email_configs:
      - to: '${ALERT_EMAIL}'
        send_resolved: true       # Notify when alert resolves
        headers:
          subject: '[NewsHub] {{ .Status | toUpper }}: {{ .CommonAnnotations.summary }}'

# D-24: Silences managed via Alertmanager UI at http://localhost:9093
```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
