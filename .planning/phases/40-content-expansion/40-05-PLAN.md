    Implement `apps/web/server/middleware/youtubeQuota.ts` per RESEARCH "Pattern 4" sketch verbatim (D-D1 / Q-03 ‚ÄĒ exposed as plain async function, NOT Express middleware, because the gate runs from inside `videoIndexService.ts` not from a route handler):

    ```typescript
    import { CacheService } from '../services/cacheService';
    import logger from '../utils/logger';

    const QUOTA_KEY = (date: string) => `youtube:quota:${date}`;
    export const YOUTUBE_DAILY_CAP = 50;  // D-D1 cap

    /**
     * Mirrors aiTierLimiter Redis-store pattern (rateLimiter.ts:115-174) but exposes
     * a plain async function ‚ÄĒ the quota check runs from inside videoIndexService
     * before each YouTube Data API search.list call, not from a route handler.
     *
     * Returns true if the call may proceed (quota available OR Redis down).
     * Returns false if the daily cap (50) has been hit.
     */
    export async function checkAndConsumeQuota(): Promise<boolean> {
      const cache = CacheService.getInstance();
      const client = cache.getClient();
      if (!client) {
        logger.debug('youtubeQuota skipped: Redis unavailable');
        return true; // graceful degradation per existing pattern (rateLimiter.ts:137-141)
      }
      const today = new Date().toISOString().slice(0, 10); // YYYY-MM-DD UTC
      const key = QUOTA_KEY(today);
      const count = await client.incr(key);
      if (count === 1) {
        const tomorrowMidnight = Math.floor(Date.UTC(
          new Date().getUTCFullYear(),
          new Date().getUTCMonth(),
          new Date().getUTCDate() + 1, 0, 0, 0
        ) / 1000);
        await client.expireat(key, tomorrowMidnight);
      }
      const ok = count <= YOUTUBE_DAILY_CAP;
      if (!ok) {
        logger.warn(`youtubeQuota: daily cap (${YOUTUBE_DAILY_CAP}) reached for ${today}; falling back to local-only`);
      }
      return ok;
    }
    ```

        let inserted = 0;

            const items = await yt.fetchChannelRSS(ch.channelId!);
            // Dedupe vs existing rows

              where: { youtubeId: { in: items.map(i => i.youtubeId) } },

            const newItems = items.filter(i => !existingIds.has(i.youtubeId));

                  description: i.description,

    import { useEffect, useState } from 'react';
    import { Play } from 'lucide-react';
    import { cn } from '../../lib/utils';

    interface Props { vimeoId: string; className?: string; }
    interface OEmbed { thumbnail_url: string; title: string; }

    const CACHE_KEY = (id: string) => `vimeo:oembed:${id}`;

    function readCache(id: string): OEmbed | null {
      if (typeof window === 'undefined') return null;
      try {
        const raw = window.localStorage.getItem(CACHE_KEY(id));
        return raw ? (JSON.parse(raw) as OEmbed) : null;
      } catch { return null; }
    }
    function writeCache(id: string, value: OEmbed): void {
      if (typeof window === 'undefined') return;
      try { window.localStorage.setItem(CACHE_KEY(id), JSON.stringify(value)); } catch { /* quota exhausted */ }
    }

    export function LiteVimeoEmbed({ vimeoId, className }: Props) {
      const [meta, setMeta] = useState<OEmbed | null>(() => readCache(vimeoId));
      const [activated, setActivated] = useState(false);

          .then((data: OEmbed) => {
            if (cancelled) return;
            setMeta(data);
            writeCache(vimeoId, data);
          })
          .catch(() => {
            if (!cancelled) setMeta({ thumbnail_url: '', title: 'Vimeo Video' });
          });
        return () => { cancelled = true; };
      }, [vimeoId, meta]);

          <iframe
            src={`https://player.vimeo.com/video/${encodeURIComponent(vimeoId)}?autoplay=1`}

            title={meta?.title ?? 'Vimeo video'}
          />
        );

      return (
        <button

    import { LiteYouTubeEmbed } from './LiteYouTubeEmbed';
    import { LiteVimeoEmbed } from './LiteVimeoEmbed';

    type Provider = 'youtube' | 'vimeo';

    interface PropsByUrl {
      url: string;
      title?: string;
      className?: string;
    }
    interface PropsById {
      provider: Provider;
      videoId: string;
      title?: string;
      className?: string;
    }
    type Props = PropsByUrl | PropsById;

    /** Returns { provider, id } or null if URL unrecognized. */
    function parseVideoUrl(url: string): { provider: Provider; id: string } | null {
      try {
        const u = new URL(url);
        // youtube.com/watch?v=ID
        if (u.hostname.includes('youtube.com') && u.searchParams.get('v')) {
          return { provider: 'youtube', id: u.searchParams.get('v')! };
        }
        // youtu.be/ID
        if (u.hostname === 'youtu.be') {
          const id = u.pathname.replace(/^\//, '');
          if (id) return { provider: 'youtube', id };
        }
        // vimeo.com/ID
        if (u.hostname.includes('vimeo.com')) {
          const m = u.pathname.match(/^\/(\d+)/);
          if (m) return { provider: 'vimeo', id: m[1] };
        }
      } catch { /* invalid URL */ }
      return null;
    }

    export function EmbeddedVideo(props: Props) {
      let provider: Provider;
      let videoId: string;

      if ('url' in props) {
        const parsed = parseVideoUrl(props.url);
        if (!parsed) {
          console.warn('EmbeddedVideo: unrecognized URL', props.url);
          return null;
        }
        provider = parsed.provider;
        videoId = parsed.id;
      } else {
        provider = props.provider;
        videoId = props.videoId;
      }

      const title = props.title ?? '';
      const className = props.className;

      return provider === 'youtube'
        ? <LiteYouTubeEmbed videoId={videoId} title={title} className={className} />
        : <LiteVimeoEmbed vimeoId={videoId} className={className} />;
    }
