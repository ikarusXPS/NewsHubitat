```typescript
type TargetLang = 'de' | 'en';
type Provider = 'deepl' | 'google' | 'libretranslate' | 'claude';

interface TranslationResult {
  text: string;
  detectedSourceLang?: string;
  provider: Provider;
  cached: boolean;
  quality: number; // 0-1 confidence score
}

interface CacheEntry {
  text: string;
  provider: Provider;
  timestamp: number;
}

const CACHE_TTL = 7 * 24 * 60 * 60 * 1000; // 7 days
const MONTHLY_LIMIT = 500000; // DeepL Free tier chars

export class TranslationService {
  static getInstance(): TranslationService;
  async translate(text: string, targetLang: TargetLang, sourceLang?: string): Promise<TranslationResult>;
  async translateBatch(texts: string[], targetLang: TargetLang): Promise<TranslationResult[]>;
  detectLanguage(text: string): string;
  getUsageStats(): { used: number; limit: number; percentage: number; providers: Provider[] };
  getCacheSize(): number;
  clearCache(): void;
}
```

Provider initialization order (from constructor):
1. DeepL (if DEEPL_API_KEY set) -> deepl-node Translator
2. Claude (if ANTHROPIC_API_KEY set) -> @anthropic-ai/sdk
3. Google (if GOOGLE_TRANSLATE_API_KEY set) -> REST API via fetch
4. LibreTranslate (always) -> REST API via fetch

Provider quality scores:
- DeepL: 0.95

- Claude: 0.85
