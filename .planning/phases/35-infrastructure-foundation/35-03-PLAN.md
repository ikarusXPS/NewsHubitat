```typescript
import { OpenAPIRegistry, OpenApiGeneratorV31 } from '@asteasolutions/zod-to-openapi';
import fs from 'fs/promises';
import path from 'path';
import {
  NewsArticleSchema,
  NewsListResponseSchema,
  NewsDetailResponseSchema,
  GeoEventsResponseSchema,
  SentimentStatsResponseSchema,
  ErrorResponseSchema,
  NewsQuerySchema,
} from './schemas';

export async function generateOpenApiSpec(outputPath: string): Promise<void> {
  const registry = new OpenAPIRegistry();

  // Register security scheme (D-06: X-API-Key)
  registry.registerComponent('securitySchemes', 'ApiKeyAuth', {
    type: 'apiKey',
    in: 'header',
    name: 'X-API-Key',
    description: 'API key with format: nh_live_xxxx or nh_test_xxxx',
  });

  // Register GET /api/v1/public/news
  registry.registerPath({
    method: 'get',
    path: '/api/v1/public/news',
    summary: 'List news articles',
    description: 'Retrieve paginated list of news articles with optional filters',
    tags: ['News'],
    security: [{ ApiKeyAuth: [] }],
    request: {
      query: NewsQuerySchema,
    },
    responses: {
      200: {
        description: 'Success',
        content: {
          'application/json': {
            schema: NewsListResponseSchema,
          },
        },
      },
      401: {
        description: 'Unauthorized - Invalid or missing API key',
        content: {
          'application/json': {
            schema: ErrorResponseSchema,
          },
        },
      },
      429: {
        description: 'Rate limit exceeded',
        content: {
          'application/json': {
            schema: ErrorResponseSchema,
          },
        },
        headers: {
          'RateLimit-Limit': {
            description: 'Maximum requests per window',
            schema: { type: 'integer', example: 10 },
          },
          'RateLimit-Remaining': {
            description: 'Remaining requests in window',
            schema: { type: 'integer', example: 0 },
          },
          'RateLimit-Reset': {
            description: 'Seconds until window resets',
            schema: { type: 'integer', example: 45 },
          },
          'Retry-After': {
            description: 'Seconds to wait before retry',
            schema: { type: 'integer', example: 45 },
          },
        },
      },
    },
  });

  // Register GET /api/v1/public/news/:id
  registry.registerPath({
    method: 'get',
    path: '/api/v1/public/news/{id}',
    summary: 'Get single news article',
    description: 'Retrieve detailed information for a specific article',
    tags: ['News'],
    security: [{ ApiKeyAuth: [] }],
    request: {
      params: z.object({
        id: z.string().openapi({ description: 'Article ID' }),
      }),
    },
    responses: {
      200: {
        description: 'Success',
        content: {
          'application/json': {
            schema: NewsDetailResponseSchema,
          },
        },
      },
      404: {
        description: 'Article not found',
        content: {
          'application/json': {
            schema: ErrorResponseSchema,
          },
        },
      },
    },
  });

  // Register GET /api/v1/public/events
  registry.registerPath({
    method: 'get',
    path: '/api/v1/public/events',
    summary: 'List geo-located events',
    description: 'Retrieve all geo-located events for map visualization',
    tags: ['Events'],
    security: [{ ApiKeyAuth: [] }],
    responses: {
      200: {
        description: 'Success',
        content: {
          'application/json': {
            schema: GeoEventsResponseSchema,
          },
        },
      },
    },
  });

  // Register GET /api/v1/public/sentiment
  registry.registerPath({
    method: 'get',
    path: '/api/v1/public/sentiment',
    summary: 'Get sentiment statistics',
    description: 'Retrieve sentiment distribution by region',
    tags: ['Analytics'],
    security: [{ ApiKeyAuth: [] }],
    responses: {
      200: {
        description: 'Success',
        content: {
          'application/json': {
            schema: SentimentStatsResponseSchema,
          },
        },
      },
    },
  });

  // Generate spec
  const generator = new OpenApiGeneratorV31(registry.definitions);
  const spec = generator.generateDocument({
    openapi: '3.1.0',
    info: {
      title: 'NewsHub Public API',
      version: '1.0.0',
      description: 'Multi-perspective global news analysis API. Access news articles, events, and analytics programmatically.',
      contact: {
        name: 'NewsHub API Support',
        url: 'https://newshub.example.com/developers',
      },
      license: {
        name: 'Proprietary',
      },
    },
    servers: [
      { url: 'https://api.newshub.example.com', description: 'Production' },
      { url: 'http://localhost:3001', description: 'Development' },
    ],
    tags: [
      { name: 'News', description: 'News article endpoints' },
      { name: 'Events', description: 'Geo-located event endpoints' },
      { name: 'Analytics', description: 'Statistics and analytics' },
    ],
  });

  // Write to file
  await fs.mkdir(path.dirname(outputPath), { recursive: true });
  await fs.writeFile(outputPath, JSON.stringify(spec, null, 2));
  console.log(`OpenAPI spec generated: ${outputPath}`);
}

// CLI execution (npm run openapi:generate)
if (import.meta.url === `file://${process.argv[1]}`) {
  const outputPath = process.argv[2] || 'public/openapi.json';
  generateOpenApiSpec(outputPath).catch(console.error);
}
```

```typescript
import { Request, Response, NextFunction } from 'express';
import { ApiKeyService } from '../services/apiKeyService';
import { CacheService } from '../services/cacheService';
import type { ApiKeyTier, ApiKeyEnv } from '../services/apiKeyService';

// Extended Request type with API key metadata
export interface ApiKeyRequest extends Request {
  apiKey?: {
    keyId: string;
    userId: string;
    tier: ApiKeyTier;
    environment: ApiKeyEnv;
  };
}

/**
 * API Key Authentication Middleware (D-06)
 * Validates X-API-Key header and attaches metadata to request
 */
export async function apiKeyAuth(
  req: ApiKeyRequest,
  res: Response,
  next: NextFunction
): Promise<void> {
  const key = req.headers['x-api-key'] as string;

  if (!key) {
    res.status(401).json({
      success: false,
      error: 'Missing X-API-Key header',
    });
    return;
  }

  try {
    const apiKeyService = ApiKeyService.getInstance();
    const cacheService = CacheService.getInstance();

    // Try cache first (5-min TTL)
    const cacheKey = `apikey:${key.substring(0, 15)}`;  // Don't cache full key
    let keyData = await cacheService.get(cacheKey);

    if (!keyData) {
      // Validate against database
      keyData = await apiKeyService.validateApiKey(key);

      if (keyData) {
        // Cache for 5 minutes
        await cacheService.set(cacheKey, keyData, 300);
      }
    }

    if (!keyData) {
      res.status(401).json({
        success: false,
        error: 'Invalid or revoked API key',
      });
      return;
    }

    // Attach to request for downstream middleware
    req.apiKey = keyData;

    // Async track usage (fire-and-forget)
    void apiKeyService.trackUsage(keyData.keyId).catch((err) => {
      console.error('Failed to track API usage:', err);
    });

    next();
  } catch (error) {
    console.error('API key auth error:', error);
    res.status(500).json({
      success: false,
      error: 'Internal server error',
    });
  }
}
```

**apps/web/server/middleware/apiKeyRateLimiter.ts** (new file):
```typescript
import { rateLimit, type RateLimitRequestHandler, type Options } from 'express-rate-limit';
import { RedisStore, type RedisReply } from 'rate-limit-redis';
import { CacheService } from '../services/cacheService';
import type { ApiKeyRequest } from './apiKeyAuth';
import type { ApiKeyTier } from '../services/apiKeyService';
import logger from '../utils/logger';

// Tier limits per D-15
const TIER_LIMITS: Record<ApiKeyTier, number> = {
  free: 10,   // 10 req/min
  pro: 100,   // 100 req/min
};

/**
 * Create tiered rate limiter for API keys (D-12, D-13, D-14, D-15)
 * Uses Redis sliding window algorithm with tier-based limits
 */
export function createApiKeyLimiter(): RateLimitRequestHandler {
  const cacheService = CacheService.getInstance();
  const redisClient = cacheService.getClient();

  // Build Redis store
  let store: RedisStore | undefined;
  if (redisClient) {
    store = new RedisStore({
      sendCommand: (command: string, ...args: string[]) =>
        redisClient.call(command, ...args) as Promise<RedisReply>,
      prefix: 'rl:api:',  // Separate namespace from internal rate limits
    });
  }

  const options: Partial<Options> = {
    windowMs: 60_000,  // 1 minute

    // Dynamic max based on tier (D-12)
    max: async (req: ApiKeyRequest) => {
      const tier = req.apiKey?.tier || 'free';
      return TIER_LIMITS[tier];
    },

    // Key by API key ID (not IP)
    keyGenerator: (req: ApiKeyRequest) => {
      return req.apiKey?.keyId || 'anonymous';
    },

    // Use Redis store (D-14: sliding window)
    store,

    // D-13: IETF RateLimit headers
    standardHeaders: true,   // RateLimit-Limit, RateLimit-Remaining, RateLimit-Reset
    legacyHeaders: false,    // No X-RateLimit-* headers

    // Custom handler with tier info
    handler: (req: ApiKeyRequest, res) => {
      const tier = req.apiKey?.tier || 'free';
      const limit = TIER_LIMITS[tier];
      const retryAfter = Math.ceil(60);  // 1 minute window

      res.set('Retry-After', String(retryAfter));
      res.status(429).json({
        success: false,
        error: 'Rate limit exceeded',
        tier,
        limit,
        window: '1 minute',
        upgradeUrl: '/pricing',
      });
    },

    // Graceful degradation if Redis unavailable
    skip: () => {
      if (!cacheService.isAvailable()) {
        logger.debug('API rate limiting skipped: Redis unavailable');
        return true;
      }
      return false;
    },
  };

  return rateLimit(options);
}
```

```typescript
import { Router, Response } from 'express';
import type { ApiKeyRequest } from '../middleware/apiKeyAuth';
import { NewsAggregator } from '../services/newsAggregator';
import { EventsService } from '../services/eventsService';
import type { PerspectiveRegion, Sentiment } from '@newshub/types';

export const publicApiRoutes = Router();

// GET /api/v1/public/news - List articles
publicApiRoutes.get('/news', async (req: ApiKeyRequest, res: Response) => {
  try {
    // Parse query parameters
    const regions = req.query.regions
      ? (req.query.regions as string).split(',') as PerspectiveRegion[]
      : undefined;
    const topics = req.query.topics
      ? (req.query.topics as string).split(',')
      : undefined;
    const sentiment = req.query.sentiment as Sentiment | undefined;
    const search = req.query.search as string | undefined;
    const page = parseInt(req.query.page as string) || 1;
    const limit = Math.min(parseInt(req.query.limit as string) || 20, 100);  // Max 100
    const offset = (page - 1) * limit;

    const newsAggregator = NewsAggregator.getInstance();
    const { articles, total } = await newsAggregator.getArticles({
      regions,
      topics,
      sentiment,
      search,
      limit,
      offset,
    });

    // Cache for 5 minutes (news changes frequently)
    res.set('Cache-Control', 'public, max-age=300');
    res.set('Vary', 'Accept-Encoding');

    res.json({
      success: true,
      data: articles,
      meta: {
        total,
        page,
        limit,
        hasMore: offset + limit < total,
      },
    });
  } catch (error) {
    console.error('Public API /news error:', error);
    res.status(500).json({
      success: false,
      error: 'Failed to fetch articles',
    });
  }
});

// GET /api/v1/public/news/:id - Single article
publicApiRoutes.get('/news/:id', async (req: ApiKeyRequest, res: Response) => {
  try {
    const newsAggregator = NewsAggregator.getInstance();
    const article = await newsAggregator.getArticleById(req.params.id);

    if (!article) {
      res.status(404).json({
        success: false,
        error: 'Article not found',
      });
      return;
    }

    // Cache for 1 hour (individual articles rarely change)
    res.set('Cache-Control', 'public, max-age=3600');

    res.json({
      success: true,
      data: article,
    });
  } catch (error) {
    console.error('Public API /news/:id error:', error);
    res.status(500).json({
      success: false,
      error: 'Failed to fetch article',
    });
  }
});

// GET /api/v1/public/events - Geo-located events
publicApiRoutes.get('/events', async (req: ApiKeyRequest, res: Response) => {
  try {
    const eventsService = EventsService.getInstance();
    const events = await eventsService.getGeoEvents();

    // Cache for 15 minutes (events update less frequently)
    res.set('Cache-Control', 'public, max-age=900');

    res.json({
      success: true,
      data: events,
      meta: {
        total: events.length,
      },
    });
  } catch (error) {
    console.error('Public API /events error:', error);
    res.status(500).json({
      success: false,
      error: 'Failed to fetch events',
    });
  }
});

// GET /api/v1/public/sentiment - Sentiment statistics
publicApiRoutes.get('/sentiment', async (req: ApiKeyRequest, res: Response) => {
  try {
    const newsAggregator = NewsAggregator.getInstance();
    const stats = await newsAggregator.getSentimentStats();

    // Cache for 10 minutes
    res.set('Cache-Control', 'public, max-age=600');

    res.json({
      success: true,
      data: stats,
    });
  } catch (error) {
    console.error('Public API /sentiment error:', error);
    res.status(500).json({
      success: false,
      error: 'Failed to fetch sentiment statistics',
    });
  }
});
```
