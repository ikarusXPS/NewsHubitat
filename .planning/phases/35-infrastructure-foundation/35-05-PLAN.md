```typescript
import { test, expect, type Page } from '@playwright/test';

// Test user credentials (use existing setup if available)
const TEST_USER = {
  email: 'publicapi-test@example.com',
  password: 'SecureTestPassword123!',
  name: 'API Test User',
};

let authToken: string;
let testApiKey: string;

test.describe('Public API', () => {
  test.beforeAll(async ({ browser }) => {
    // Create test user and get auth token
    const page = await browser.newPage();

    // Register user (skip if already exists)
    await page.goto('/');
    await page.click('text=Sign Up');
    await page.fill('input[type="email"]', TEST_USER.email);
    await page.fill('input[name="name"]', TEST_USER.name);
    await page.fill('input[type="password"]', TEST_USER.password);
    await page.click('button[type="submit"]');

    // Wait for redirect and extract token from localStorage
    await page.waitForTimeout(2000);
    authToken = await page.evaluate(() => localStorage.getItem('newshub-auth-token') || '');

    await page.close();
  });

  test.describe('API Key Management', () => {
    test('should create API key via UI', async ({ page }) => {
      await page.goto('/developers');

      // Wait for page load
      await expect(page.locator('h1')).toContainText('Developer API');

      // Fill key name
      await page.fill('input[placeholder*="Key name"]', 'Test Key');

      // Click create
      await page.click('button:has-text("Create Key")');

      // Wait for key to appear
      await expect(page.locator('code:has-text("nh_live_")')).toBeVisible();

      // Extract key from UI
      const keyElement = page.locator('code:has-text("nh_live_")');
      testApiKey = await keyElement.textContent() || '';

      expect(testApiKey).toMatch(/^nh_live_[0-9A-Za-z]{24}_[0-9A-F]{4}$/);

      // Copy key
      await page.click('button:has-text("Copy")');

      // Dismiss key display
      await page.click('text=I\'ve copied my key');

      // Verify key appears in list
      await expect(page.locator('text=Test Key')).toBeVisible();
    });

    test('should enforce 3-key limit', async ({ page }) => {
      await page.goto('/developers');

      // Create 3 keys
      for (let i = 1; i <= 3; i++) {
        await page.fill('input[placeholder*="Key name"]', `Key ${i}`);
        await page.click('button:has-text("Create Key")');
        await page.waitForTimeout(500);
        if (i < 3) {
          await page.click('text=I\'ve copied my key');
        }
      }

      // 4th key should be disabled
      await expect(page.locator('button:has-text("Create Key")')).toBeDisabled();

      // Clean up: revoke all keys
      const revokeButtons = page.locator('button[title="Revoke API key"]');
      const count = await revokeButtons.count();
      for (let i = 0; i < count; i++) {
        await revokeButtons.first().click();
        await page.click('text=OK');  // Confirm dialog
        await page.waitForTimeout(300);
      }
    });
  });

  test.describe('API Authentication', () => {
    test('should allow access with valid API key', async ({ request }) => {
      const response = await request.get('/api/v1/public/news', {
        headers: {
          'X-API-Key': testApiKey,
        },
      });

      expect(response.status()).toBe(200);

      const data = await response.json();
      expect(data.success).toBe(true);
      expect(Array.isArray(data.data)).toBe(true);
      expect(data.meta).toHaveProperty('total');
      expect(data.meta).toHaveProperty('page');
      expect(data.meta).toHaveProperty('limit');
    });

    test('should return 401 for missing API key', async ({ request }) => {
      const response = await request.get('/api/v1/public/news');

      expect(response.status()).toBe(401);

      const data = await response.json();
      expect(data.success).toBe(false);
      expect(data.error).toContain('Missing X-API-Key');
    });

    test('should return 401 for invalid API key', async ({ request }) => {
      const response = await request.get('/api/v1/public/news', {
        headers: {
          'X-API-Key': 'invalid-key-format',
        },
      });

      expect(response.status()).toBe(401);

      const data = await response.json();
      expect(data.success).toBe(false);
      expect(data.error).toContain('Invalid');
    });

    test('should return 401 for revoked API key', async ({ page, request }) => {
      // Create a key via UI
      await page.goto('/developers');
      await page.fill('input[placeholder*="Key name"]', 'Revoke Test Key');
      await page.click('button:has-text("Create Key")');

      const keyElement = page.locator('code:has-text("nh_live_")');
      const revokeTestKey = await keyElement.textContent() || '';
      await page.click('text=I\'ve copied my key');

      // Verify it works
      let response = await request.get('/api/v1/public/news', {
        headers: { 'X-API-Key': revokeTestKey },
      });
      expect(response.status()).toBe(200);

      // Revoke it
      await page.click('button[title="Revoke API key"]');
      await page.click('text=OK');  // Confirm
      await page.waitForTimeout(500);

      // Verify it's revoked
      response = await request.get('/api/v1/public/news', {
        headers: { 'X-API-Key': revokeTestKey },
      });
      expect(response.status()).toBe(401);
    });
  });

  test.describe('Rate Limiting', () => {
    test('should enforce free tier rate limit (10 req/min)', async ({ request }) => {
      // Make 10 requests (should succeed)
      for (let i = 0; i < 10; i++) {
        const response = await request.get('/api/v1/public/news?limit=1', {
          headers: { 'X-API-Key': testApiKey },
        });
        expect(response.status()).toBe(200);
      }

      // 11th request should be rate limited
      const response = await request.get('/api/v1/public/news?limit=1', {
        headers: { 'X-API-Key': testApiKey },
      });

      expect(response.status()).toBe(429);

      const data = await response.json();
      expect(data.success).toBe(false);
      expect(data.error).toContain('Rate limit exceeded');
      expect(data.tier).toBe('free');
      expect(data.limit).toBe(10);
    });

    test('should include rate limit headers', async ({ request }) => {
      const response = await request.get('/api/v1/public/news', {
        headers: { 'X-API-Key': testApiKey },
      });

      // IETF RateLimit headers
      expect(response.headers()['ratelimit-limit']).toBeDefined();
      expect(response.headers()['ratelimit-remaining']).toBeDefined();
      expect(response.headers()['ratelimit-reset']).toBeDefined();
    });

    test('should include Retry-After header on 429', async ({ request }) => {
      // Exhaust rate limit
      for (let i = 0; i < 10; i++) {
        await request.get('/api/v1/public/news?limit=1', {
          headers: { 'X-API-Key': testApiKey },
        });
      }

      // Get 429 response
      const response = await request.get('/api/v1/public/news', {
        headers: { 'X-API-Key': testApiKey },
      });

      expect(response.status()).toBe(429);
      expect(response.headers()['retry-after']).toBeDefined();

      const retryAfter = parseInt(response.headers()['retry-after']);
      expect(retryAfter).toBeGreaterThan(0);
      expect(retryAfter).toBeLessThanOrEqual(60);
    });
  });

  test.describe('Public API Endpoints', () => {
    test('should return news articles with filters', async ({ request }) => {
      const response = await request.get('/api/v1/public/news?regions=usa&limit=5', {
        headers: { 'X-API-Key': testApiKey },
      });

      expect(response.status()).toBe(200);

      const data = await response.json();
      expect(data.success).toBe(true);
      expect(Array.isArray(data.data)).toBe(true);
      expect(data.data.length).toBeLessThanOrEqual(5);

      // Verify article schema
      if (data.data.length > 0) {
        const article = data.data[0];
        expect(article).toHaveProperty('id');
        expect(article).toHaveProperty('title');
        expect(article).toHaveProperty('content');
        expect(article).toHaveProperty('url');
        expect(article).toHaveProperty('sourceId');
        expect(article).toHaveProperty('region');
        expect(article).toHaveProperty('sentiment');
        expect(article).toHaveProperty('publishedAt');
      }
    });

    test('should return single article by ID', async ({ request }) => {
      // Get first article from list
      const listResponse = await request.get('/api/v1/public/news?limit=1', {
        headers: { 'X-API-Key': testApiKey },
      });
      const listData = await listResponse.json();

      if (listData.data.length === 0) {
        test.skip();  // No articles available
        return;
      }

      const articleId = listData.data[0].id;

      // Fetch individual article
      const response = await request.get(`/api/v1/public/news/${articleId}`, {
        headers: { 'X-API-Key': testApiKey },
      });

      expect(response.status()).toBe(200);

      const data = await response.json();
      expect(data.success).toBe(true);
      expect(data.data.id).toBe(articleId);
    });

    test('should return 404 for non-existent article', async ({ request }) => {
      const response = await request.get('/api/v1/public/news/non-existent-id', {
        headers: { 'X-API-Key': testApiKey },
      });

      expect(response.status()).toBe(404);

      const data = await response.json();
      expect(data.success).toBe(false);
      expect(data.error).toContain('not found');
    });

    test('should return geo events', async ({ request }) => {
      const response = await request.get('/api/v1/public/events', {
        headers: { 'X-API-Key': testApiKey },
      });

      expect(response.status()).toBe(200);

      const data = await response.json();
      expect(data.success).toBe(true);
      expect(Array.isArray(data.data)).toBe(true);
      expect(data.meta).toHaveProperty('total');
    });

    test('should return sentiment statistics', async ({ request }) => {
      const response = await request.get('/api/v1/public/sentiment', {
        headers: { 'X-API-Key': testApiKey },
      });

      expect(response.status()).toBe(200);

      const data = await response.json();
      expect(data.success).toBe(true);
      expect(Array.isArray(data.data)).toBe(true);

      // Verify stats schema
      if (data.data.length > 0) {
        const stat = data.data[0];
        expect(stat).toHaveProperty('region');
        expect(stat).toHaveProperty('positive');
        expect(stat).toHaveProperty('negative');
        expect(stat).toHaveProperty('neutral');
        expect(stat).toHaveProperty('total');
      }
    });

    test('should serve OpenAPI spec without auth', async ({ request }) => {
      const response = await request.get('/api/openapi.json');

      expect(response.status()).toBe(200);

      const spec = await response.json();
      expect(spec.openapi).toBe('3.1.0');
      expect(spec.info.title).toContain('NewsHub');
      expect(spec.paths).toHaveProperty('/api/v1/public/news');
    });
  });

  test.describe('Cache Headers', () => {
    test('should set Cache-Control headers on news list', async ({ request }) => {
      const response = await request.get('/api/v1/public/news', {
        headers: { 'X-API-Key': testApiKey },
      });

      expect(response.status()).toBe(200);
      expect(response.headers()['cache-control']).toContain('max-age=300');
    });

    test('should set Cache-Control headers on single article', async ({ request }) => {
      const listResponse = await request.get('/api/v1/public/news?limit=1', {
        headers: { 'X-API-Key': testApiKey },
      });
      const listData = await listResponse.json();

      if (listData.data.length === 0) {
        test.skip();
        return;
      }

      const response = await request.get(`/api/v1/public/news/${listData.data[0].id}`, {
        headers: { 'X-API-Key': testApiKey },
      });

      expect(response.status()).toBe(200);
      expect(response.headers()['cache-control']).toContain('max-age=3600');
    });
  });
});
```
