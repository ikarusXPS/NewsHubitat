```prisma
// Stores hashed keys with tier-based rate limits
model ApiKey {
  id          String   @id @default(cuid())
  keyHash     String   @unique  // bcrypt hash of full key (never store plaintext)
  name        String              // User-defined name (e.g., "Production", "Staging")
  tier        ApiKeyTier @default(free)
  environment ApiKeyEnv  @default(live)

  // Ownership and limits
  userId      String
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  // Usage tracking
  createdAt   DateTime @default(now())
  lastUsedAt  DateTime?
  requestCount Int     @default(0)  // Monthly request counter (reset by cleanup job)

  // Revocation
  revokedAt   DateTime?
  revokedReason String?

  // Indexes for lookup and cleanup
  @@index([userId])
  @@index([revokedAt])
  @@index([lastUsedAt])
  @@index([createdAt])
}

enum ApiKeyTier {
  free  // 10 req/min per D-15
  pro   // 100 req/min per D-15
}

enum ApiKeyEnv {
  live  // Production keys (nh_live_)
  test  // Test/sandbox keys (nh_test_)
}
```

```typescript
import { customAlphabet } from 'nanoid';
import crypto from 'crypto';
import bcrypt from 'bcryptjs';
import { prisma } from '../db/prisma';

const nanoid = customAlphabet('0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz', 24);

export type ApiKeyTier = 'free' | 'pro';
export type ApiKeyEnv = 'live' | 'test';

export interface ApiKeyComponents {
  prefix: string;  // nh_live or nh_test
  random: string;  // 24-char nanoid
  checksum: string;  // 4-char SHA-256 hash
  full: string;  // prefix_random_checksum
}

export interface ApiKeyData {
  id: string;
  name: string;
  tier: ApiKeyTier;
  environment: ApiKeyEnv;
  createdAt: Date;
  lastUsedAt: Date | null;
  requestCount: number;
}

const MAX_KEYS_PER_USER = 3;  // D-10

export class ApiKeyService {
  private static instance: ApiKeyService;

  private constructor() {
    console.log('API Key service initialized');
  }

  static getInstance(): ApiKeyService {
    if (!ApiKeyService.instance) {
      ApiKeyService.instance = new ApiKeyService();
    }
    return ApiKeyService.instance;
  }

  /**
   * Generate API key with format: nh_{env}_{random}_{checksum}
   * Example: nh_live_1A2b3C4d5E6f7G8h9I0j1K2L_A3B4
   */
  generateApiKey(environment: ApiKeyEnv = 'live'): ApiKeyComponents {
    const prefix = `nh_${environment}`;
    const random = nanoid();  // 24 chars

    // Generate checksum: first 4 chars of SHA-256 hash
    const hash = crypto.createHash('sha256').update(`${prefix}_${random}`).digest('hex');
    const checksum = hash.substring(0, 4).toUpperCase();

    const full = `${prefix}_${random}_${checksum}`;

    return { prefix, random, checksum, full };
  }

  /**
   * Validate API key format and checksum before database lookup
   * Returns false for invalid format to prevent unnecessary DB queries
   */
  validateApiKeyFormat(key: string): boolean {
    // Format: nh_(live|test)_<24-chars>_<4-chars>
    const pattern = /^nh_(live|test)_[0-9A-Za-z]{24}_[0-9A-F]{4}$/;
    if (!pattern.test(key)) {
      return false;
    }

    // Extract components
    const parts = key.split('_');
    const prefix = `${parts[0]}_${parts[1]}`;  // nh_live or nh_test
    const random = parts[2];
    const providedChecksum = parts[3];

    // Recalculate checksum
    const hash = crypto.createHash('sha256').update(`${prefix}_${random}`).digest('hex');
    const expectedChecksum = hash.substring(0, 4).toUpperCase();

    return providedChecksum === expectedChecksum;
  }

  /**
   * Create new API key for user
   * Throws if user already has MAX_KEYS_PER_USER active keys
   */
  async createApiKey(
    userId: string,
    name: string,
    tier: ApiKeyTier = 'free',
    environment: ApiKeyEnv = 'live'
  ): Promise<{ key: string; keyData: ApiKeyData }> {
    // Check key limit (D-10)
    const activeKeys = await prisma.apiKey.count({
      where: {
        userId,
        revokedAt: null,
      },
    });

    if (activeKeys >= MAX_KEYS_PER_USER) {
      throw new Error(`Maximum ${MAX_KEYS_PER_USER} API keys per user. Revoke an existing key first.`);
    }

    // Generate key
    const { full } = this.generateApiKey(environment);

    // Hash for storage (never store plaintext)
    const keyHash = await bcrypt.hash(full, 10);

    // Create in database
    const apiKey = await prisma.apiKey.create({
      data: {
        keyHash,
        name,
        tier,
        environment,
        userId,
        createdAt: new Date(),
      },
    });

    // Return plaintext key ONCE (user must copy it now)
    const keyData: ApiKeyData = {
      id: apiKey.id,
      name: apiKey.name,
      tier: apiKey.tier as ApiKeyTier,
      environment: apiKey.environment as ApiKeyEnv,
      createdAt: apiKey.createdAt,
      lastUsedAt: apiKey.lastUsedAt,
      requestCount: apiKey.requestCount,
    };

    return { key: full, keyData };
  }

  /**
   * Validate API key against database
   * Returns key metadata if valid, null if invalid/revoked
   */
  async validateApiKey(key: string): Promise<{
    keyId: string;
    userId: string;
    tier: ApiKeyTier;
    environment: ApiKeyEnv;
  } | null> {
    // Pre-validation: check format and checksum
    if (!this.validateApiKeyFormat(key)) {
      return null;
    }

    // Fetch all active keys (need to iterate for bcrypt comparison)
    const apiKeys = await prisma.apiKey.findMany({
      where: { revokedAt: null },
      select: {
        id: true,
        keyHash: true,
        userId: true,
        tier: true,
        environment: true,
      },
    });

    // Find matching key by hash comparison
    for (const record of apiKeys) {
      const match = await bcrypt.compare(key, record.keyHash);
      if (match) {
        return {
          keyId: record.id,
          userId: record.userId,
          tier: record.tier as ApiKeyTier,
          environment: record.environment as ApiKeyEnv,
        };
      }
    }

    return null;
  }

  /**
   * Get user's API keys (returns metadata only, never hashes)
   */
  async getUserApiKeys(userId: string): Promise<ApiKeyData[]> {
    const keys = await prisma.apiKey.findMany({
      where: { userId, revokedAt: null },
      orderBy: { createdAt: 'desc' },
    });

    return keys.map((key) => ({
      id: key.id,
      name: key.name,
      tier: key.tier as ApiKeyTier,
      environment: key.environment as ApiKeyEnv,
      createdAt: key.createdAt,
      lastUsedAt: key.lastUsedAt,
      requestCount: key.requestCount,
    }));
  }

  /**
   * Revoke API key
   */
  async revokeApiKey(keyId: string, userId: string, reason?: string): Promise<void> {
    const key = await prisma.apiKey.findUnique({
      where: { id: keyId },
    });

    if (!key) {
      throw new Error('API key not found');
    }

    if (key.userId !== userId) {
      throw new Error('Unauthorized: not your API key');
    }

    if (key.revokedAt) {
      throw new Error('API key already revoked');
    }

    await prisma.apiKey.update({
      where: { id: keyId },
      data: {
        revokedAt: new Date(),
        revokedReason: reason,
      },
    });
  }

  /**
   * Update lastUsedAt timestamp (fire-and-forget)
   * Called async from middleware, never blocks request
   */
  async trackUsage(keyId: string): Promise<void> {
    await prisma.apiKey.update({
      where: { id: keyId },
      data: {
        lastUsedAt: new Date(),
        requestCount: { increment: 1 },
      },
    });
  }
}

// Export singleton getter for convenience
export const getApiKeyService = () => ApiKeyService.getInstance();
```

```typescript
import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { ApiKeyService } from './apiKeyService';
import { prisma } from '../db/prisma';
import bcrypt from 'bcryptjs';

describe('ApiKeyService', () => {
  let service: ApiKeyService;
  let testUserId: string;

  beforeEach(async () => {
    service = ApiKeyService.getInstance();

    // Create test user
    const user = await prisma.user.create({
      data: {
        email: 'apitest@example.com',
        passwordHash: await bcrypt.hash('password123', 10),
        name: 'API Test User',
        emailVerified: true,
      },
    });
    testUserId = user.id;
  });

  afterEach(async () => {
    // Clean up
    await prisma.apiKey.deleteMany({ where: { userId: testUserId } });
    await prisma.user.delete({ where: { id: testUserId } });
  });

  describe('generateApiKey', () => {
    it('should generate key with live environment prefix', () => {
      const { prefix, random, checksum, full } = service.generateApiKey('live');

      expect(prefix).toBe('nh_live');
      expect(random).toHaveLength(24);
      expect(checksum).toHaveLength(4);
      expect(full).toMatch(/^nh_live_[0-9A-Za-z]{24}_[0-9A-F]{4}$/);
    });

    it('should generate key with test environment prefix', () => {
      const { prefix, full } = service.generateApiKey('test');

      expect(prefix).toBe('nh_test');
      expect(full).toMatch(/^nh_test_[0-9A-Za-z]{24}_[0-9A-F]{4}$/);
    });

    it('should generate unique keys on multiple calls', () => {
      const key1 = service.generateApiKey('live');
      const key2 = service.generateApiKey('live');

      expect(key1.full).not.toBe(key2.full);
    });
  });

  describe('validateApiKeyFormat', () => {
    it('should return true for valid key with correct checksum', () => {
      const { full } = service.generateApiKey('live');
      expect(service.validateApiKeyFormat(full)).toBe(true);
    });

    it('should return false for key with wrong checksum', () => {
      const { prefix, random } = service.generateApiKey('live');
      const invalidKey = `${prefix}_${random}_ZZZZ`;

      expect(service.validateApiKeyFormat(invalidKey)).toBe(false);
    });

    it('should return false for malformed key (wrong format)', () => {
      expect(service.validateApiKeyFormat('invalid-key')).toBe(false);
      expect(service.validateApiKeyFormat('nh_live_short')).toBe(false);
      expect(service.validateApiKeyFormat('nh_invalid_1234567890123456789012_ABCD')).toBe(false);
    });

    it('should return false for key with lowercase checksum', () => {
      const { prefix, random } = service.generateApiKey('live');
      const key = `${prefix}_${random}_abcd`;  // checksum must be uppercase

      expect(service.validateApiKeyFormat(key)).toBe(false);
    });
  });

  describe('createApiKey', () => {
    it('should create API key and return plaintext', async () => {
      const result = await service.createApiKey(testUserId, 'Test Key', 'free', 'live');

      expect(result.key).toMatch(/^nh_live_[0-9A-Za-z]{24}_[0-9A-F]{4}$/);
      expect(result.keyData.name).toBe('Test Key');
      expect(result.keyData.tier).toBe('free');
      expect(result.keyData.environment).toBe('live');

    it('should hash key before storing in database', async () => {
      const result = await service.createApiKey(testUserId, 'Test Key', 'free', 'live');

      const dbKey = await prisma.apiKey.findUnique({
        where: { id: result.keyData.id },
      });

      expect(dbKey!.keyHash).not.toBe(result.key);
      expect(dbKey!.keyHash).toHaveLength(60);  // bcrypt hash length

      // Verify hash matches
      const match = await bcrypt.compare(result.key, dbKey!.keyHash);
      expect(match).toBe(true);
    });

    it('should throw error when user has 3 active keys', async () => {

      await service.createApiKey(testUserId, 'Key 1', 'free', 'live');
      await service.createApiKey(testUserId, 'Key 2', 'free', 'live');
      await service.createApiKey(testUserId, 'Key 3', 'free', 'live');

      // 4th key should fail

    it('should allow creating new key after revoking one', async () => {
      // Create 3 keys
      const key1 = await service.createApiKey(testUserId, 'Key 1', 'free', 'live');
      await service.createApiKey(testUserId, 'Key 2', 'free', 'live');
      await service.createApiKey(testUserId, 'Key 3', 'free', 'live');

      await service.revokeApiKey(key1.keyData.id, testUserId);

      // Should allow creating new key

      expect(key4.keyData.name).toBe('Key 4');
    });
  });

  describe('validateApiKey', () => {

      expect(result).toBeNull();

    it('should return key metadata for valid key', async () => {
      const created = await service.createApiKey(testUserId, 'Test Key', 'pro', 'live');
      const validated = await service.validateApiKey(created.key);

      expect(validated).not.toBeNull();
      expect(validated!.keyId).toBe(created.keyData.id);
      expect(validated!.userId).toBe(testUserId);
      expect(validated!.tier).toBe('pro');
      expect(validated!.environment).toBe('live');
    });

    it('should return null for revoked key', async () => {
      const created = await service.createApiKey(testUserId, 'Test Key', 'free', 'live');
      await service.revokeApiKey(created.keyData.id, testUserId);

      const validated = await service.validateApiKey(created.key);
      expect(validated).toBeNull();
    });

    it('should return null for key with wrong checksum', async () => {
      const { prefix, random } = service.generateApiKey('live');
      const invalidKey = `${prefix}_${random}_ZZZZ`;

      const result = await service.validateApiKey(invalidKey);
      expect(result).toBeNull();

      expect(keys).toEqual([]);

      await service.createApiKey(testUserId, 'Key 1', 'free', 'live');
      await service.createApiKey(testUserId, 'Key 2', 'pro', 'test');

      const keys = await service.getUserApiKeys(testUserId);

      expect(keys).toHaveLength(2);
      expect(keys[0].name).toBe('Key 2');  // Most recent first
      expect(keys[1].name).toBe('Key 1');
    });

    it('should not return revoked keys', async () => {
      const key1 = await service.createApiKey(testUserId, 'Key 1', 'free', 'live');
      await service.createApiKey(testUserId, 'Key 2', 'free', 'live');

      await service.revokeApiKey(key1.keyData.id, testUserId);

      const keys = await service.getUserApiKeys(testUserId);
      expect(keys).toHaveLength(1);
      expect(keys[0].name).toBe('Key 2');
    });
  });

  describe('revokeApiKey', () => {
    it('should set revokedAt timestamp', async () => {
      const created = await service.createApiKey(testUserId, 'Test Key', 'free', 'live');
      await service.revokeApiKey(created.keyData.id, testUserId);

      const dbKey = await prisma.apiKey.findUnique({
        where: { id: created.keyData.id },
      });

      expect(dbKey!.revokedAt).not.toBeNull();
      expect(dbKey!.revokedAt).toBeInstanceOf(Date);
    });

    it('should throw error for non-existent key', async () => {
      await expect(
        service.revokeApiKey('non-existent-id', testUserId)
      ).rejects.toThrow('API key not found');
    });

    it('should throw error when revoking another user\'s key', async () => {
      const otherUser = await prisma.user.create({
        data: {
          email: 'other@example.com',
          passwordHash: await bcrypt.hash('password123', 10),
          name: 'Other User',
          emailVerified: true,
```
