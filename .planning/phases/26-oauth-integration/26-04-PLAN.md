```typescript
const { t } = useTranslation('auth');
const { loginWithOAuth } = useAuth();

// OAuth state
const [linkingProvider, setLinkingProvider] = useState<'google' | 'github' | null>(null);
const [linkingEmail, setLinkingEmail] = useState<string>('');
const [oauthLoading, setOauthLoading] = useState<'google' | 'github' | null>(null);

const { openOAuthPopup, error: oauthError } = useOAuthPopup({
  onSuccess: async (result) => {
    setOauthLoading(null);
    if (result.token) {
      try {
        await loginWithOAuth(result.token);
        onClose();
        // Reset form
        setEmail('');
        setPassword('');
        setName('');
      } catch (err) {
        setError(err instanceof Error ? err.message : 'OAuth login failed');
      }
    }
  },
  onError: (errorMsg) => {
    setOauthLoading(null);
    setError(errorMsg);
  },
  onNeedsLinking: (email) => {
    setOauthLoading(null);
    setLinkingEmail(email);
    setLinkingProvider(oauthLoading);
  },
});

const handleGoogleAuth = () => {
  setError(null);
  setOauthLoading('google');
  openOAuthPopup('google');
};

const handleGitHubAuth = () => {
  setError(null);
  setOauthLoading('github');
  openOAuthPopup('github');
};

const handleLinkAccount = async (password: string) => {
  if (!linkingProvider) return { success: false, error: 'No provider selected' };

  try {
    const response = await fetch('/api/auth/oauth/link', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        provider: linkingProvider,
        providerId: '', // Will be populated from OAuth flow state
        email: linkingEmail,
        password,
      }),
    });

    const data = await response.json();

    if (!data.success) {
      return { success: false, error: data.error };
    }

    // Login with the returned token
    await loginWithOAuth(data.data.token);
    setLinkingProvider(null);
    setLinkingEmail('');
    onClose();
    return { success: true };
  } catch (err) {
    return { success: false, error: err instanceof Error ? err.message : 'Link failed' };
  }
};
```

```typescript
interface ConnectedProvider {
  google: boolean;
  github: boolean;
  hasPassword: boolean;
}

export function ConnectedAccounts() {
  const { t } = useTranslation(['settings', 'auth']);
  const { token, loginWithOAuth } = useAuth();

  const [providers, setProviders] = useState<ConnectedProvider | null>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [unlinkingProvider, setUnlinkingProvider] = useState<'google' | 'github' | null>(null);
  const [connectingProvider, setConnectingProvider] = useState<'google' | 'github' | null>(null);
  const [unlinkPassword, setUnlinkPassword] = useState('');

  // Toast state
  const [toast, setToast] = useState<{ message: string; type: 'success' | 'error' | 'info'; isOpen: boolean }>({
    message: '',
    type: 'info',
    isOpen: false,
  });

  // Confirm dialog state
  const [confirmDialog, setConfirmDialog] = useState<{
    isOpen: boolean;
    provider: 'google' | 'github' | null;
  }>({
    isOpen: false,
    provider: null,
  });

  // Fetch connected providers
  useEffect(() => {
    const fetchProviders = async () => {
      if (!token) return;

      try {
        const response = await fetch('/api/auth/oauth/providers', {
          headers: { Authorization: `Bearer ${token}` },
        });

        if (response.ok) {
          const data = await response.json();
          setProviders(data.data);
        }
      } catch (err) {
        console.error('Failed to fetch providers:', err);
      } finally {
        setIsLoading(false);
      }
    };

    fetchProviders();
  }, [token]);

  const { openOAuthPopup } = useOAuthPopup({
    onSuccess: async (result) => {
      setConnectingProvider(null);
      if (result.token) {
        // Re-fetch providers after connecting
        try {
          const response = await fetch('/api/auth/oauth/providers', {
            headers: { Authorization: `Bearer ${token}` },
          });
          if (response.ok) {
            const data = await response.json();
            setProviders(data.data);
          }
          setToast({
            message: t('auth:oauth.linkSuccess', { provider: connectingProvider }),
            type: 'success',
            isOpen: true,
          });
        } catch {
          // Ignore
        }
      }
    },
    onError: (errorMsg) => {
      setConnectingProvider(null);
      setToast({ message: errorMsg, type: 'error', isOpen: true });
    },
    onNeedsLinking: () => {
      setConnectingProvider(null);
      setToast({
        message: t('auth:oauth.emailMismatch'),
        type: 'error',
        isOpen: true,
      });
    },
  });

  const handleConnect = (provider: 'google' | 'github') => {
    setConnectingProvider(provider);
    openOAuthPopup(provider);
  };

  const handleUnlink = async () => {
    if (!confirmDialog.provider) return;

    // D-12: Check if this is the only login method
    const otherProvider = confirmDialog.provider === 'google' ? 'github' : 'google';
    const hasOtherOAuth = providers?.[otherProvider];
    const hasPassword = providers?.hasPassword;

    if (!hasPassword && !hasOtherOAuth) {
      setToast({
        message: t('auth:oauth.unlinkBlocked'),
        type: 'error',
        isOpen: true,
      });
      setConfirmDialog({ isOpen: false, provider: null });
      return;
    }

    setUnlinkingProvider(confirmDialog.provider);

    try {
      const response = await fetch('/api/auth/oauth/unlink', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${token}`,
        },
        body: JSON.stringify({
          provider: confirmDialog.provider,
          password: unlinkPassword,
        }),
      });

      const data = await response.json();

      if (!data.success) {
        setToast({ message: data.error, type: 'error', isOpen: true });
        return;
      }

      // Update local state
      setProviders((prev) =>
        prev
          ? { ...prev, [confirmDialog.provider!]: false }
          : prev
      );
      setToast({
        message: t('auth:oauth.unlinkSuccess', { provider: confirmDialog.provider }),
        type: 'success',
        isOpen: true,
      });
    } catch (err) {
      setToast({
        message: err instanceof Error ? err.message : 'Unlink failed',
        type: 'error',
        isOpen: true,
      });
    } finally {
      setUnlinkingProvider(null);
      setUnlinkPassword('');
      setConfirmDialog({ isOpen: false, provider: null });
    }
  };

  if (isLoading) {
    return (
      <div className="rounded-lg border border-gray-700 bg-gray-800 p-4">
        <div className="flex items-center justify-center py-8">
          <Loader2 className="h-6 w-6 animate-spin text-gray-400" />
        </div>
      </div>
    );
  }

  const providerList = [
    { id: 'google' as const, name: 'Google', icon: GoogleIcon, connected: providers?.google },
    { id: 'github' as const, name: 'GitHub', icon: GitHubIcon, connected: providers?.github },
  ];

  return (
    <>
      <div className="rounded-lg border border-gray-700 bg-gray-800 p-4">
        <h2 className="text-lg font-medium text-white mb-4 flex items-center gap-2">
          <Link2 className="h-5 w-5 text-[#00f0ff]" />
          {t('settings:connectedAccounts.title')}
        </h2>

        <div className="space-y-3">
          {providerList.map(({ id, name, icon: Icon, connected }) => (
            <div
              key={id}
              className="flex items-center justify-between p-3 rounded-lg bg-gray-700/50"
            >
              <div className="flex items-center gap-3">
                <Icon />
                <div>
                  <p className="text-white font-medium">{name}</p>
                  <p className={cn(
                    'text-xs flex items-center gap-1',
                    connected ? 'text-green-400' : 'text-gray-500'
                  )}>
                    {connected ? (
                      <>
                        <Check className="h-3 w-3" />
                        {t('auth:oauth.connected')}
                      </>
                    ) : (
                      t('auth:oauth.notConnected')
                    )}
                  </p>
                </div>
              </div>

              {connected ? (
                <button
                  onClick={() => setConfirmDialog({ isOpen: true, provider: id })}
                  disabled={unlinkingProvider === id}
                  className={cn(
                    'rounded-lg px-3 py-1.5 text-sm border transition-colors',
                    'border-[#ff0044]/30 text-[#ff0044]',
                    'hover:bg-[#ff0044]/10',
                    unlinkingProvider === id && 'opacity-50 cursor-not-allowed'
                  )}
                >
                  {unlinkingProvider === id ? (
                    <Loader2 className="h-4 w-4 animate-spin" />
                  ) : (
                    t('auth:oauth.unlink')
                  )}
                </button>
              ) : (
                <button
                  onClick={() => handleConnect(id)}
                  disabled={connectingProvider === id}
                  className={cn(
                    'rounded-lg px-3 py-1.5 text-sm border transition-colors',
                    'border-[#00f0ff]/30 text-[#00f0ff]',
                    'hover:bg-[#00f0ff]/10',
                    connectingProvider === id && 'opacity-50 cursor-not-allowed'
                  )}
                >
                  {connectingProvider === id ? (
                    <Loader2 className="h-4 w-4 animate-spin" />
                  ) : (
                    t('auth:oauth.connect')
                  )}
                </button>
              )}
            </div>
          ))}
        </div>

        {!providers?.hasPassword && (
          <p className="mt-4 text-xs text-gray-500">
            {t('settings:connectedAccounts.noPasswordHint')}
          </p>
        )}
      </div>

      {/* Unlink Confirmation Dialog */}
      <ConfirmDialog
        isOpen={confirmDialog.isOpen}
        title={t('settings:connectedAccounts.unlinkTitle', { provider: confirmDialog.provider })}
        message={t('settings:connectedAccounts.unlinkMessage', { provider: confirmDialog.provider })}
        onConfirm={handleUnlink}
        onCancel={() => {
          setConfirmDialog({ isOpen: false, provider: null });
          setUnlinkPassword('');
        }}
        variant="warning"
      />

      {/* Toast */}
      <Toast
        message={toast.message}
        type={toast.type}
        isOpen={toast.isOpen}
        onClose={() => setToast({ ...toast, isOpen: false })}
      />
    </>
  );
}
```
