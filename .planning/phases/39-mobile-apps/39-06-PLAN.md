#   - certificates (readonly): pull .p12 + .mobileprovision from match git repo
#   - build_release: archive + export .ipa for App Store distribution

default_platform(:ios)

platform :ios do
  before_all do

    %w[
      MATCH_GIT_URL
      MATCH_PASSWORD
      MATCH_KEYCHAIN_PASSWORD
      APPLE_TEAM_ID

      readonly: true,
      keychain_name:     "fastlane_tmp_keychain",
      keychain_password: ENV["MATCH_KEYCHAIN_PASSWORD"]
    )
  end

  desc "Build a signed .ipa for App Store distribution (TestFlight upload deferred to operator)"
  lane :build_release do
    certificates
    # gym = build_app; emits .ipa under apps/mobile/ios/build/
    build_app(
      workspace:           "App/App.xcworkspace",  # cap add ios generated this path
      scheme:              "App",
      configuration:       "Release",
      export_method:       "app-store",
      output_directory:    "./build",

    )
  end
end

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GitHub Actions runner ↔ repo secrets | Secrets injected via `${{ secrets.X }}`; runner job has access to plaintext for the duration of the job |
| Runner workspace ↔ artifact upload | `actions/upload-artifact@v4` archives anything in the configured `path:` |
| Local dev machine ↔ keystore.jks | Operator generates keystore once; loss = permanent app brick |
| Fastlane runner ↔ match git repo | Read-only mode (`readonly: true`) on CI; only the operator's local Mac has write access during `match init` |
| AAB / IPA artifact ↔ store ingestion | Manual operator step; out of CI scope |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-39-06-01 | Information Disclosure | keystore.properties accidentally committed | mitigate | .gitignore (Task 5) covers the file at every depth; commit-time sanity check in Task 9 step 2 explicitly aborts if the file appears in git status |
| T-39-06-02 | Information Disclosure | release.keystore accidentally committed | mitigate | Same: .gitignore + Task 9 abort check; Android workflow's Cleanup step also rm -fs the file post-build |
| T-39-06-03 | Information Disclosure | .p8 App Store Connect key written to runner home dir leaks via artifact upload | mitigate | Upload step's path is scoped to apps/mobile/ios/build/NewsHub.ipa ONLY; ~/.appstoreconnect/ is outside that path and never archived |
| T-39-06-04 | Tampering | Forked PR triggers signed build via push trigger | mitigate | Workflows are workflow_dispatch only; PRs CANNOT trigger them. Required-secrets are also gated to default branch by GitHub's secret access policy |
| T-39-06-05 | Spoofing | Malicious actor publishes a fake .ipa pretending to be NewsHub | accept | Out of CI scope — Apple App Review + bundle ID uniqueness handle this. Apple won't accept duplicate bundle IDs. |
| T-39-06-06 | Repudiation | Build operator uploads a binary that doesn't match the source | mitigate | github.run_number embedded in artifact name; `git log` for the commit SHA is auditable; CI step logs persist 30d (and artifacts 30d) |
| T-39-06-07 | Denial of Service | Malicious workflow_dispatch from compromised maintainer account drains macOS runner minutes | accept | Repo is private; maintainer access is the trust boundary. GitHub Actions usage limits provide a soft cap. |
| T-39-06-08 | Elevation of Privilege | Stale Fastlane match cert grants signing access after rotation | mitigate | match readonly mode + match's documented rotation flow (`match nuke distribution` + re-init); operator follows runbook on cert expiry |
| T-39-06-09 | Tampering | Android AAB shipped without our signature (debug-signed) | mitigate | Workflow's `jarsigner -verify` step explicitly checks the AAB is signed; build fails if not |
| T-39-06-10 | Information Disclosure | macOS runner's temp keychain persists between jobs leaking certs | mitigate | Fastfile uses `setup_ci` with a uniquely-named temp keychain (`fastlane_tmp_keychain`); GitHub macos-latest runners are ephemeral (destroyed after each job), so cross-job leak is structurally impossible |
