2. **Modify `apps/web/src/hooks/useHapticFeedback.ts`** — preserve the existing API (`lightTap`, `mediumTap`, `heavyTap`, `selectionChanged`, `notificationOccurred`). Add Capacitor haptics via dynamic import so web bundles don't try to resolve native plugin code:

```typescript
import { isNativeApp } from '../lib/platform';

async function tryNativeHaptics(style: 'LIGHT' | 'MEDIUM' | 'HEAVY' | 'SELECTION' | 'SUCCESS' | 'WARNING' | 'ERROR'): Promise<boolean> {
  if (!isNativeApp()) return false;
  try {
    const { Haptics, ImpactStyle, NotificationType } = await import('@capacitor/haptics');
    if (style === 'SELECTION') {
      await Haptics.selectionChanged();
    } else if (style === 'SUCCESS' || style === 'WARNING' || style === 'ERROR') {
      await Haptics.notification({ type: NotificationType[style] });
    } else {
      await Haptics.impact({ style: ImpactStyle[style as 'LIGHT' | 'MEDIUM' | 'HEAVY'] });
    }
    return true;
  } catch {
    return false;  // plugin not loaded or call failed — fall back to web vibrate
  }
}

export function useHapticFeedback() {
  const lightTap = async () => {
    if (await tryNativeHaptics('LIGHT')) return;
    if ('vibrate' in navigator) navigator.vibrate(10);
  };
  const mediumTap = async () => {
    if (await tryNativeHaptics('MEDIUM')) return;
    if ('vibrate' in navigator) navigator.vibrate(20);
  };
  const heavyTap = async () => {
    if (await tryNativeHaptics('HEAVY')) return;
    if ('vibrate' in navigator) navigator.vibrate([30, 20, 30]);
  };
  const selectionChanged = async () => {
    if (await tryNativeHaptics('SELECTION')) return;
    if ('vibrate' in navigator) navigator.vibrate(5);
  };
  const notificationOccurred = async (type: 'success' | 'warning' | 'error' = 'success') => {
    const styleMap = { success: 'SUCCESS', warning: 'WARNING', error: 'ERROR' } as const;
    if (await tryNativeHaptics(styleMap[type])) return;
    if ('vibrate' in navigator) {
      const pattern = type === 'success' ? [10, 30, 10] : type === 'warning' ? [20, 50, 20] : [50, 100, 50];
      navigator.vibrate(pattern);
    }
  };
  return { lightTap, mediumTap, heavyTap, selectionChanged, notificationOccurred };
```
