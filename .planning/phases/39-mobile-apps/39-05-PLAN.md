<objective>
Add biometric login (D-11) — JWT in iOS Keychain / Android Keystore via @capgo/capacitor-native-biometric. First login still uses password; subsequent app opens prompt biometric, decrypt the JWT, populate localStorage, and let `AuthContext`'s mount-time `verifyToken` pick it up. 3-fail fallback to password (the Capgo plugin's `negativeButtonText` lets the user opt out per attempt). Settings has an enable/disable toggle. Includes the mandatory Pitfall A1 spike to verify the Capgo plugin marks credentials as device-only (no iCloud Keychain sync).

Purpose: D-11 says biometric is a login REPLACEMENT (not just a gate over an already-loaded session). The implementation requires careful coordination between the bootstrap path (main.tsx, before React mounts), AuthContext (writes JWT), and Settings UI (user-controllable toggle). The A1 spike is risk mitigation: if the Capgo plugin doesn't set device-only, JWTs sync via iCloud and break the per-device security guarantee.

Output: Working biometric flow on iOS/Android. User logs in once with email+password; on next app open, Face ID / Touch ID / fingerprint unlocks the JWT and the user is signed in instantly. User can disable biometric from Settings, which deletes the keychain entry. Web is unaffected (every biometric call early-returns).
</objective>

// apps/web/src/lib/biometric.ts (NEW)
import { NativeBiometric } from '@capgo/capacitor-native-biometric';
import { isNativeApp } from './platform';

const BIO_SERVER = 'newshub-jwt';

export async function persistJwtBehindBiometric(jwt: string, userId: string): Promise<void> {
  if (!isNativeApp()) return;
  try {
    const { isAvailable } = await NativeBiometric.isAvailable();
    if (!isAvailable) return;
    await NativeBiometric.setCredentials({ username: userId, password: jwt, server: BIO_SERVER });
  } catch (err) {
    console.warn('[biometric] persist failed', err);
  }
}

export async function unlockJwtWithBiometric(): Promise<string | null> {
  if (!isNativeApp()) return null;
  try {
    const { isAvailable } = await NativeBiometric.isAvailable();
    if (!isAvailable) return null;
    await NativeBiometric.verifyIdentity({
      reason: 'Use Face ID to sign in to NewsHub',
      title: 'Sign in',
      negativeButtonText: 'Use password',  // D-11 fallback
    });
    const creds = await NativeBiometric.getCredentials({ server: BIO_SERVER });
    return creds.password;
  } catch {
    return null;  // user cancelled OR biometry failed → fall through to password
  }
}

export async function clearBiometricJwt(): Promise<void> {
  if (!isNativeApp()) return;
  try {
    await NativeBiometric.deleteCredentials({ server: BIO_SERVER });
  } catch {
    /* idempotent */
  }
}

export async function isBiometricAvailable(): Promise<boolean> {
  if (!isNativeApp()) return false;
  try {
    const { isAvailable } = await NativeBiometric.isAvailable();
    return !!isAvailable;
  } catch {
    return false;
  }
}

export async function hasStoredCredential(): Promise<boolean> {
  if (!isNativeApp()) return false;
  try {
    await NativeBiometric.getCredentials({ server: BIO_SERVER });
    // NOTE: this prompts biometry — only call when user explicitly clicks "Check status"
    return true;
  } catch {
    return false;
  }
}

export function useBiometricAuth() {

import { Toast } from '../Toast';

const BIO_FLAG_KEY = 'newshub-biometric-enabled';

export function BiometricSection() {
  const { t } = useTranslation('settings');
  const { isAvailable, enable, disable } = useBiometricAuth();
  const [enabled, setEnabled] = useState<boolean>(() => localStorage.getItem(BIO_FLAG_KEY) === '1');
  const [toast, setToast] = useState<{ message: string; type: 'success' | 'error' | 'info'; isOpen: boolean }>({ message: '', type: 'info', isOpen: false });

  // Hide entirely on web — biometric is native-only
  if (!isNativeApp()) return null;

  const onToggle = async (checked: boolean) => {
    if (checked) {
      const ok = await enable();
      if (ok) {
        localStorage.setItem(BIO_FLAG_KEY, '1');
        setEnabled(true);
        setToast({ message: t('biometric.enabled'), type: 'success', isOpen: true });
      } else {
        setToast({ message: t('biometric.enableFailed'), type: 'error', isOpen: true });
      }
    } else {
      await disable();
      localStorage.removeItem(BIO_FLAG_KEY);
      setEnabled(false);
      setToast({ message: t('biometric.disabled'), type: 'info', isOpen: true });
    }
  };

  return (
    <>
      <div className="rounded-lg border border-gray-700 bg-gray-800 p-4">
        <h2 className="text-lg font-medium text-white mb-4 flex items-center gap-2">
          <Fingerprint className="h-5 w-5 text-[#00f0ff]" />
          {t('biometric.title')}
        </h2>
        {!isAvailable && (
          <p className="text-sm text-gray-400">{t('biometric.notAvailable')}</p>
        )}
        {isAvailable && (
          <label className="flex items-center justify-between py-2">
            <span className="text-sm text-gray-300">{t('biometric.useFaceId')}</span>
            <input type="checkbox" checked={enabled} onChange={e => void onToggle(e.target.checked)} />
          </label>
        )}
      </div>
      <Toast message={toast.message} type={toast.type} isOpen={toast.isOpen} onClose={() => setToast(t => ({ ...t, isOpen: false }))} />
    </>
  );
}
