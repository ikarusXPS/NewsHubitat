import { useEffect } from 'react';
import { PushNotifications } from '@capacitor/push-notifications';
import { Capacitor } from '@capacitor/core';
import { useAuth } from '../contexts/AuthContext';
import { isNativeApp } from '../lib/platform';

export function usePushNotifications() {
  const { token: jwt, isAuthenticated } = useAuth();
  useEffect(() => {
    if (!isNativeApp() || !isAuthenticated || !jwt) return;
    const init = async () => {
      let perm = await PushNotifications.checkPermissions();
      if (perm.receive === 'prompt') perm = await PushNotifications.requestPermissions();
      if (perm.receive !== 'granted') return;
      // Pitfall 3: Android 13+ POST_NOTIFICATIONS is now granted; safe to register
      await PushNotifications.register();
    };
    const cleanups: Array<() => void> = [];
    PushNotifications.addListener('registration', async (t) => {
      await fetch('/api/push/subscribe', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${jwt}` },
        body: JSON.stringify({ platform: Capacitor.getPlatform(), token: t.value }),
      });
    }).then(h => cleanups.push(() => h.remove()));
    PushNotifications.addListener('pushNotificationActionPerformed', (n) => {
      const articleId = n.notification.data?.articleId;
      if (articleId) window.location.href = `/article/${articleId}`;
    }).then(h => cleanups.push(() => h.remove()));
    init();
    return () => cleanups.forEach(c => c());
  }, [jwt, isAuthenticated]);
}

1. **Implement four trigger handlers inside `notificationFanoutService.ts`**. Replace the Plan-2 skeleton's `subscribe`/`dispatch` with the full version. The service now imports `prisma`, `PushService`, `PushVolumeService`, and a new utility for quiet-hours math.

```typescript
import { prisma } from '../db/prisma';
import { CacheService } from './cacheService';
import { PushService } from './pushService';
import { PushVolumeService } from './pushVolumeService';
import logger from '../utils/logger';
import type { PushPayload } from '@newshub/types';
import type { NewsArticle } from '../../src/types';

const COLD_START_MIN_HISTORY = 5;  // D-05

export class NotificationFanoutService {
  private static instance: NotificationFanoutService;
  private isRunning = false;
  private pushService = PushService.getInstance();
  private volumeService = PushVolumeService.getInstance();

  static getInstance(): NotificationFanoutService { /* singleton */ }

  start(): void { /* idempotent */ }
  stop(): void { /* idempotent */ }

  /** Called fire-and-forget from workerEmitter on every article emission */
  async dispatch(eventKind: 'new-article' | 'breaking', article: NewsArticle): Promise<void> {
    if (!this.isRunning || !this.pushService.isAvailable()) return;
    try {
      if (eventKind === 'breaking') {
        await this.runBreakingTrigger(article);
        await this.runAffinityTrigger(article);   // also runs on every new article — D-05 says it's not exclusive to breaking
        await this.runKeywordTrigger(article);
      } else {
        await this.runAffinityTrigger(article);
        await this.runKeywordTrigger(article);
      }
    } catch (err) {
      logger.error('[NotificationFanoutService] dispatch error', err);
    }
  }

  private async runBreakingTrigger(article: NewsArticle): Promise<void> {
    // Find all users with NotificationPreference.triggerBreaking + region match
    const recipients = await prisma.user.findMany({
      where: {
        defaultRegions: { has: article.perspective },
        notificationPreference: { is: { enabled: true, triggerBreaking: true } },
      },
      select: {
        id: true, subscriptionTier: true, timezone: true,
        notificationPreference: { select: { quietHoursStart: true, quietHoursEnd: true } },
        pushSubscriptions: { select: { token: true, platform: true } },
      },
    });
    for (const u of recipients) {
      if (!await this.allowedToSend(u, article.id, 'BREAKING')) continue;
      await this.sendToUser(u, article, 'breaking');
    }
  }

  private async runAffinityTrigger(article: NewsArticle): Promise<void> {
    // PREMIUM only (D-06)
    // Cold-start: skip users with <5 ReadingHistory entries
    // Top-N entities/topics from last 30d intersect with article.topics → match → send
    // Implementation: query ReadingHistory grouped by user, count last-30d entries; if ≥5 and any topic intersects, recipient
    const cutoff = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000);
    const candidates = await prisma.user.findMany({
      where: {
        subscriptionTier: { in: ['PREMIUM', 'ENTERPRISE'] },
        notificationPreference: { is: { enabled: true, triggerAffinity: true } },
      },
      select: {
        id: true, subscriptionTier: true, timezone: true,
        notificationPreference: { select: { quietHoursStart: true, quietHoursEnd: true } },
        pushSubscriptions: { select: { token: true, platform: true } },
      },
    });
    for (const u of candidates) {
      const hist = await prisma.readingHistory.findMany({
        where: { userId: u.id, readAt: { gte: cutoff } },
        select: { article: { select: { topics: true } } },
        take: 100,
      });
      if (hist.length < COLD_START_MIN_HISTORY) continue;  // D-05 cold start
      const topics = new Set(hist.flatMap(h => h.article?.topics ?? []));
      const overlap = (article.topics ?? []).some(t => topics.has(t));
      if (!overlap) continue;
      if (!await this.allowedToSend(u, article.id, 'AFFINITY')) continue;
      await this.sendToUser(u, article, 'affinity');
    }
  }

  private async runKeywordTrigger(article: NewsArticle): Promise<void> {
    // PREMIUM only (D-06)
    // Match article title/content against KeywordWatch.keyword (case-insensitive)
    const watches = await prisma.keywordWatch.findMany({
      include: {
        user: {
          select: {
            id: true, subscriptionTier: true, timezone: true,
            notificationPreference: { select: { enabled: true, triggerKeyword: true, quietHoursStart: true, quietHoursEnd: true } },
            pushSubscriptions: { select: { token: true, platform: true } },
          },
        },
      },
    });
    const haystack = `${article.title ?? ''} ${article.content ?? ''}`.toLowerCase();
    for (const w of watches) {
      const u = w.user;
      if (u.subscriptionTier === 'FREE') continue;  // D-06 server-side gate
      if (!u.notificationPreference?.enabled || !u.notificationPreference?.triggerKeyword) continue;
      if (!haystack.includes(w.keyword.toLowerCase())) continue;
      // Region scope: empty regionsScope = all regions; otherwise must include article.perspective
      const scope = (w.regionsScope as string[] | null) ?? [];
      if (scope.length > 0 && !scope.includes(article.perspective)) continue;
      if (!await this.allowedToSend(u, article.id, 'KEYWORD')) continue;
      await this.sendToUserWithKeyword(u, article, w.keyword);
    }
  }

  /** Called by DigestSchedulerService.registerTickHandler — sweeps users whose digestHour matches current UTC hour shifted to their timezone */
  async runDigestTick(): Promise<void> {
    const nowUtc = new Date();
    const candidates = await prisma.user.findMany({
      where: {
        subscriptionTier: { in: ['PREMIUM', 'ENTERPRISE'] },  // D-06
        notificationPreference: { is: { enabled: true, triggerDigest: true } },
      },
      select: {
        id: true, subscriptionTier: true, timezone: true,
        notificationPreference: { select: { digestHour: true, quietHoursStart: true, quietHoursEnd: true } },
        pushSubscriptions: { select: { token: true, platform: true } },
      },
    });
    for (const u of candidates) {
      const tz = u.timezone || 'UTC';
      const localHour = parseInt(new Intl.DateTimeFormat('en-US', { timeZone: tz, hour: '2-digit', hour12: false }).format(nowUtc), 10);
      if (localHour !== u.notificationPreference?.digestHour) continue;
      // Cold-start gate
      const histCount = await prisma.readingHistory.count({ where: { userId: u.id } });
      if (histCount < COLD_START_MIN_HISTORY) continue;
      const articleCount = await prisma.newsArticle.count({ where: { publishedAt: { gte: new Date(Date.now() - 24 * 60 * 60 * 1000) } } });
      if (articleCount === 0) continue;
      if (!await this.allowedToSend(u, `digest:${nowUtc.toISOString().slice(0, 10)}`, 'DIGEST')) continue;
      await this.sendDigest(u, nowUtc.toISOString().slice(0, 10), articleCount);
    }
  }

  // -- helpers ----------------------------------------------------
  private async allowedToSend(
    u: { id: string; subscriptionTier: string | null; timezone: string | null; notificationPreference?: { quietHoursStart: string; quietHoursEnd: string } | null; pushSubscriptions: { token: string }[] },
    articleId: string,
    triggerKind: 'BREAKING' | 'AFFINITY' | 'KEYWORD' | 'DIGEST'
  ): Promise<boolean> {
    if (u.pushSubscriptions.length === 0) return false;
    const tier = (u.subscriptionTier as 'FREE' | 'PREMIUM' | 'ENTERPRISE' | null) ?? 'FREE';
    // D-06 tier gate
    if (tier === 'FREE' && triggerKind !== 'BREAKING') return false;
    // D-07 dedup (per-articleId once per user)
    if (await this.volumeService.alreadySent(u.id, articleId)) return false;
    // D-07 quiet hours
    if (this.isInQuietHours(u.timezone, u.notificationPreference?.quietHoursStart ?? '22:00', u.notificationPreference?.quietHoursEnd ?? '07:00')) return false;
    // D-07 daily cap
    if (!await this.volumeService.canSend(u.id, tier)) return false;
    return true;
  }

  private isInQuietHours(timezone: string | null, startHHMM: string, endHHMM: string): boolean {
    const tz = timezone || 'UTC';
    const now = new Date();
    const localHM = new Intl.DateTimeFormat('en-US', { timeZone: tz, hour: '2-digit', minute: '2-digit', hour12: false, hourCycle: 'h23' }).format(now).replace(/\s/g, '');
    // Handles wrap-around (e.g., 22:00–07:00)
    if (startHHMM <= endHHMM) return localHM >= startHHMM && localHM < endHHMM;
    return localHM >= startHHMM || localHM < endHHMM;
  }

  private async sendToUser(u: { id: string; pushSubscriptions: { token: string }[] }, article: NewsArticle, kind: 'breaking' | 'affinity'): Promise<void> {
    const tokens = u.pushSubscriptions.map(s => s.token);
    const payload: PushPayload = {
      title: kind === 'breaking' ? `Breaking: ${article.title}` : `New: ${article.title}`,
      body: article.summary?.slice(0, 200) ?? '',
      articleId: article.id,
      deeplink: `/article/${article.id}`,
      triggerKind: kind,
      region: kind === 'breaking' ? article.perspective : undefined,
      locale: (article.language ?? 'en') as 'de' | 'en' | 'fr',
    };
    await this.pushService.sendToTokens(tokens, payload);
    await this.volumeService.markSent(u.id, article.id);
  }

  private async sendToUserWithKeyword(u: { id: string; pushSubscriptions: { token: string }[] }, article: NewsArticle, keyword: string): Promise<void> {
    const tokens = u.pushSubscriptions.map(s => s.token);
    const payload: PushPayload = {
      title: `Watch: '${keyword}'`, body: article.title ?? '',
      articleId: article.id, deeplink: `/feed?keyword=${encodeURIComponent(keyword)}`,
      triggerKind: 'keyword', keyword,
      locale: (article.language ?? 'en') as 'de' | 'en' | 'fr',
    };
    await this.pushService.sendToTokens(tokens, payload);
    await this.volumeService.markSent(u.id, article.id);
  }

  private async sendDigest(u: { id: string; pushSubscriptions: { token: string }[] }, isoDate: string, articleCount: number): Promise<void> {
    const tokens = u.pushSubscriptions.map(s => s.token);
    const payload: PushPayload = {
      title: 'Your Daily NewsHub Digest', body: `${articleCount} articles in the last 24h`,
      deeplink: `/digest/${isoDate}`,
      triggerKind: 'digest', digestDate: isoDate,
      locale: 'en',
    };
    await this.pushService.sendToTokens(tokens, payload);
    await this.volumeService.markSent(u.id, `digest:${isoDate}`);
  }
}

import { useState, useEffect, useCallback } from 'react';
import { Bell, Plus, X } from 'lucide-react';
import { useTranslation } from 'react-i18next';
import { useAuth } from '../../contexts/AuthContext';
import { isNativeApp } from '../../lib/platform';
import { Toast } from '../Toast';

interface Prefs {
  enabled: boolean;
  triggerBreaking: boolean;
  triggerAffinity: boolean;
  triggerKeyword: boolean;
  triggerDigest: boolean;
  digestHour: number;
  quietHoursStart: string;
  quietHoursEnd: string;
}

export function NotificationsSection() {
  const { t } = useTranslation(['settings', 'common']);
  const { token, user } = useAuth();
  const [prefs, setPrefs] = useState<Prefs | null>(null);
  const [keywords, setKeywords] = useState<{ id: string; keyword: string }[]>([]);
  const [newKw, setNewKw] = useState('');
  const [toast, setToast] = useState<{ message: string; type: 'success' | 'error' | 'info'; isOpen: boolean }>({ message: '', type: 'info', isOpen: false });

  const tier = (user as { subscriptionTier?: string } | null)?.subscriptionTier ?? 'FREE';
  const showKeywordUi = !isNativeApp() || tier !== 'FREE';

  useEffect(() => {
    if (!token) return;
    fetch('/api/notifications/preferences', { headers: { Authorization: `Bearer ${token}` } })
      .then(r => r.json()).then(j => j.success && setPrefs(j.data)).catch(() => {});
    if (showKeywordUi) {
      fetch('/api/keyword-watch', { headers: { Authorization: `Bearer ${token}` } })
        .then(r => r.json()).then(j => j.success && setKeywords(j.data)).catch(() => {});
    }
  }, [token, showKeywordUi]);

  const patchPrefs = useCallback(async (patch: Partial<Prefs>) => {
    if (!token) return;
    const r = await fetch('/api/notifications/preferences', {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` },
      body: JSON.stringify(patch),
    });
    const j = await r.json();
    if (j.success) setPrefs(j.data);
    else setToast({ message: j.error ?? 'Update failed', type: 'error', isOpen: true });
  }, [token]);

  const addKeyword = async () => {
    if (!token || !newKw.trim()) return;
    const r = await fetch('/api/keyword-watch', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` },
      body: JSON.stringify({ keyword: newKw.trim(), regionsScope: [] }),
    });
    const j = await r.json();
    if (j.success) { setKeywords(k => [j.data, ...k]); setNewKw(''); }
    else setToast({ message: j.error ?? 'Add failed', type: 'error', isOpen: true });
  };

  const removeKeyword = async (id: string) => {
    if (!token) return;
    await fetch(`/api/keyword-watch/${id}`, { method: 'DELETE', headers: { Authorization: `Bearer ${token}` } });
    setKeywords(k => k.filter(x => x.id !== id));
  };

  if (!prefs) return null;
  return (
    <>
      <div className="rounded-lg border border-gray-700 bg-gray-800 p-4">
        <h2 className="text-lg font-medium text-white mb-4 flex items-center gap-2">
          <Bell className="h-5 w-5 text-[#00f0ff]" />
          {t('settings:notifications.title', 'Notifications')}
        </h2>
        {/* Master toggle */}
        <label className="flex items-center justify-between py-2">
          <span className="text-sm text-gray-300">{t('settings:notifications.enabled', 'Enable notifications')}</span>
          <input type="checkbox" checked={prefs.enabled} onChange={e => patchPrefs({ enabled: e.target.checked })} />
        </label>
        {/* 4 trigger toggles */}
        <label className="flex items-center justify-between py-2"><span className="text-sm text-gray-300">{t('settings:notifications.triggerBreaking', 'Breaking news')}</span><input type="checkbox" checked={prefs.triggerBreaking} onChange={e => patchPrefs({ triggerBreaking: e.target.checked })} /></label>
        <label className="flex items-center justify-between py-2"><span className="text-sm text-gray-300">{t('settings:notifications.triggerAffinity', 'Personalized')}</span><input type="checkbox" checked={prefs.triggerAffinity} onChange={e => patchPrefs({ triggerAffinity: e.target.checked })} /></label>
        <label className="flex items-center justify-between py-2"><span className="text-sm text-gray-300">{t('settings:notifications.triggerKeyword', 'Keyword watch')}</span><input type="checkbox" checked={prefs.triggerKeyword} onChange={e => patchPrefs({ triggerKeyword: e.target.checked })} /></label>
        <label className="flex items-center justify-between py-2"><span className="text-sm text-gray-300">{t('settings:notifications.triggerDigest', 'Daily digest')}</span><input type="checkbox" checked={prefs.triggerDigest} onChange={e => patchPrefs({ triggerDigest: e.target.checked })} /></label>
        {/* Quiet hours */}
        <div className="grid grid-cols-2 gap-2 py-2">
          <label className="text-sm text-gray-300">{t('settings:notifications.quietStart', 'Quiet from')}<input type="time" value={prefs.quietHoursStart} onChange={e => patchPrefs({ quietHoursStart: e.target.value })} className="w-full bg-gray-900 border border-gray-700 rounded px-2 py-1" /></label>
          <label className="text-sm text-gray-300">{t('settings:notifications.quietEnd', 'Quiet until')}<input type="time" value={prefs.quietHoursEnd} onChange={e => patchPrefs({ quietHoursEnd: e.target.value })} className="w-full bg-gray-900 border border-gray-700 rounded px-2 py-1" /></label>
        </div>
        {/* Digest hour */}
        <label className="flex items-center justify-between py-2"><span className="text-sm text-gray-300">{t('settings:notifications.digestHour', 'Digest hour (0-23)')}</span><input type="number" min={0} max={23} value={prefs.digestHour} onChange={e => patchPrefs({ digestHour: parseInt(e.target.value, 10) })} className="w-16 bg-gray-900 border border-gray-700 rounded px-2 py-1" /></label>
        {/* Keyword watch */}
        {showKeywordUi ? (
          <details className="mt-4">
            <summary className="cursor-pointer text-sm text-[#00f0ff]">{t('settings:notifications.keywordWatch', 'Keyword Watch')} ({keywords.length})</summary>
            <div className="mt-2 space-y-1">
              {keywords.map(k => (
                <div key={k.id} className="flex items-center justify-between bg-gray-900 px-2 py-1 rounded">
                  <span className="text-sm text-gray-300">{k.keyword}</span>
                  <button onClick={() => removeKeyword(k.id)} aria-label="remove"><X className="h-4 w-4 text-gray-500" /></button>
                </div>
              ))}
              <div className="flex gap-1">
                <input value={newKw} onChange={e => setNewKw(e.target.value)} placeholder={t('settings:notifications.addKeyword', 'Add keyword')} className="flex-1 bg-gray-900 border border-gray-700 rounded px-2 py-1 text-sm" />
                <button onClick={addKeyword} className="px-2 py-1 bg-[#00f0ff]/10 border border-[#00f0ff]/30 text-[#00f0ff] rounded"><Plus className="h-4 w-4" /></button>
              </div>
            </div>
          </details>
        ) : (
          <div className="mt-4 px-3 py-2 text-xs text-gray-300 rounded-lg border border-gray-700 bg-gray-800/60">
            <p>{t('common:upgrade.notAvailableOnPlan')}</p>
            <p className="text-gray-400 mt-1">{t('common:upgrade.visitWebsiteHint', { domain: 'newshub.example' })}</p>
          </div>
        )}
      </div>
      <Toast message={toast.message} type={toast.type} isOpen={toast.isOpen} onClose={() => setToast(t => ({ ...t, isOpen: false }))} />
    </>
  );
}
```
