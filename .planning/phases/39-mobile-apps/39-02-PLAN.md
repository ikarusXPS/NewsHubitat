```typescript
const TOKEN_DEAD_CODES = new Set([
  'messaging/registration-token-not-registered',
  'messaging/invalid-registration-token',
  'messaging/invalid-argument',
]);
async sendToTokens(tokens: string[], payload: PushPayload): Promise<void> {
  if (tokens.length === 0) return;
  if (!this.app) return;  // env-not-configured graceful no-op
  const chunks: string[][] = [];
  for (let i = 0; i < tokens.length; i += 200) chunks.push(tokens.slice(i, i + 200));
  for (const chunk of chunks) {
    const res = await admin.messaging(this.app).sendEachForMulticast({
      tokens: chunk,
      notification: { title: payload.title, body: payload.body },
      data: {
        articleId: payload.articleId ?? '',
        deeplink: payload.deeplink,
        triggerKind: payload.triggerKind,
        keyword: payload.keyword ?? '',
        region: payload.region ?? '',
        digestDate: payload.digestDate ?? '',
        locale: payload.locale,
      },
      apns: { payload: { aps: { sound: 'default', 'mutable-content': 1, category: payload.triggerKind } } },
      android: { priority: 'high', notification: { channelId: this.channelFor(payload.triggerKind) } },
    });
    const dead: string[] = [];
    res.responses.forEach((r, idx) => {
      if (!r.success && r.error && TOKEN_DEAD_CODES.has(r.error.code)) dead.push(chunk[idx]);
    });
    if (dead.length) {
      await prisma.pushSubscription.deleteMany({ where: { token: { in: dead } } });
      logger.info(`[PushService] cleaned up ${dead.length} dead tokens`);
    }
  }
}
private channelFor(kind: PushTriggerKind): string {
  return kind === 'breaking' ? 'breaking' : kind === 'digest' ? 'digest' : 'personalized';
}
```

```typescript
const FREE_DAILY_CAP = 5;       // D-07
const PREMIUM_DAILY_CAP = 10;   // D-07
const DEDUP_TTL_SECONDS = 7 * 24 * 60 * 60;  // D-07: 7-day TTL

export class PushVolumeService {
  private static instance: PushVolumeService;
  private cache = CacheService.getInstance();

  static getInstance(): PushVolumeService {
    if (!PushVolumeService.instance) PushVolumeService.instance = new PushVolumeService();
    return PushVolumeService.instance;

  /** D-07 daily cap — must be called BEFORE pushService.sendToTokens for a given user */
  async canSend(userId: string, tier: 'FREE' | 'PREMIUM' | 'ENTERPRISE'): Promise<boolean> {
    const cap = tier === 'FREE' ? FREE_DAILY_CAP : PREMIUM_DAILY_CAP;
    const today = new Date().toISOString().slice(0, 10);  // YYYY-MM-DD UTC
    const key = `push:count:${userId}:${today}`;
    const client = this.cache.getClient();
    if (!client) return true;  // Redis unavailable — fail open in dev (matches aiTierLimiter skip)
    const count = await client.incr(key);
    if (count === 1) await client.expire(key, 24 * 60 * 60);  // 24h TTL on first incr
    if (count > cap) {
      // Decrement back so we don't penalize this user further today
      await client.decr(key);
      logger.info(`[PushVolumeService] cap reached userId=${userId} tier=${tier} count=${count - 1}/${cap}`);
      return false;
    }
    return true;
  }

  /** D-07 dedup — record that this user has been sent this article (across all triggers) */
  async markSent(userId: string, articleId: string): Promise<void> {
    const client = this.cache.getClient();
    if (!client) return;
    const key = `push:sent:${userId}:${articleId}`;
    await client.set(key, '1', 'EX', DEDUP_TTL_SECONDS);
  }

  /** D-07 dedup — same articleId can fire only once per user across all four triggers */
  async alreadySent(userId: string, articleId: string): Promise<boolean> {
    const client = this.cache.getClient();
    if (!client) return false;
    const key = `push:sent:${userId}:${articleId}`;
    const exists = await client.exists(key);
    return exists > 0;
  }
```

```typescript
const HOUR_MS = 60 * 60 * 1000;

export class DigestSchedulerService {
  private static instance: DigestSchedulerService;
  private intervalId: NodeJS.Timeout | null = null;
  private isRunning = false;
  private tickHandler: (() => Promise<void>) | null = null;

    if (!DigestSchedulerService.instance) DigestSchedulerService.instance = new DigestSchedulerService();
    return DigestSchedulerService.instance;

  /** Plan 4 calls registerTickHandler() to subscribe its digest-fanout handler */
  registerTickHandler(handler: () => Promise<void>): void {
    this.tickHandler = handler;
  }

  start(): void {
    if (this.isRunning) { logger.warn('Digest scheduler already running'); return; }
    this.isRunning = true;
    this.intervalId = setInterval(() => {
      void this.runTick();
    }, HOUR_MS);
    logger.info('[DigestSchedulerService] started — hourly tick');
  }

  stop(): void {
    if (this.intervalId) clearInterval(this.intervalId);
    this.intervalId = null;
    this.isRunning = false;
    logger.info('[DigestSchedulerService] stopped');
  }

  private async runTick(): Promise<void> {
    try {
      if (this.tickHandler) {
        await this.tickHandler();
      } else {
        logger.debug('[DigestSchedulerService] tick — no handler registered yet (Plan 4 pending)');
      }
    } catch (err) {
      logger.error('[DigestSchedulerService] tick error', err);
    }
  }
```
