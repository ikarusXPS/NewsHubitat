```typescript
import { Database, Clock } from 'lucide-react';
import { cn } from '../lib/utils';

interface CacheIndicatorProps {
  isFromCache: boolean;
  cacheAge: number | null;
  className?: string;
}

function formatCacheAge(ageMs: number): string {
  const minutes = Math.floor(ageMs / 60000);
  const hours = Math.floor(minutes / 60);

  if (hours > 0) {
    return `${hours}h ${minutes % 60}m ago`;
  }
  if (minutes > 0) {
    return `${minutes}m ago`;
  }
  return 'Just now';
}

export function CacheIndicator({ isFromCache, cacheAge, className }: CacheIndicatorProps) {
  if (!isFromCache) return null;

  return (
    <div className={cn('flex items-center gap-2 px-3 py-2 rounded-lg', className)}>
      <Database className="h-4 w-4 text-[#ffcc00]" />
      <span className="text-xs font-mono text-[#ffcc00]">Cached Data</span>
      {cacheAge !== null && (
        <div className="flex items-center gap-1 text-xs font-mono text-gray-400">
          <Clock className="h-3 w-3" />
          {formatCacheAge(cacheAge)}
        </div>
      )}
    </div>
  );
}
```

```typescript
import { useQuery } from '@tanstack/react-query';
import { cacheService } from '../services/cacheService';

export function useCachedQuery<T>({ queryKey, queryFn, cacheKey, ttl }: Options<T>) {
  return useQuery({
    queryKey,
    queryFn: async () => {
      try {
        const response = await queryFn();
        await cacheService.setArticles(cacheKey, response, ttl);
        return response;
      } catch (error) {
        const cached = await cacheService.getArticles(cacheKey);
        if (cached) return cached;
        throw error;
      }
    },
  });
}
```

```typescript
export function useCachedQuery<T>({ queryKey, queryFn, cacheKey, ttl }: Options<T>) {
  const [cacheAge, setCacheAge] = React.useState<number | null>(null);
  const [isFromCache, setIsFromCache] = React.useState(false);

  const query = useQuery({
    queryKey,
    queryFn: async () => {
      try {
        const response = await queryFn();
        await cacheService.setArticles(cacheKey, response, ttl);
        setIsFromCache(false);
        setCacheAge(null);
        return response;
      } catch (error) {
        const cached = await cacheService.getArticles(cacheKey);
        if (cached) {
          const metadata = await cacheService.getMetadata(cacheKey);
          if (metadata) {
            setIsFromCache(true);
            setCacheAge(Date.now() - metadata.timestamp);
          }
          return cached;
        }
        throw error;
      }
    },
  });

  return {
    ...query,
    cacheAge,
    isFromCache,
  };
}
```

**Note:** Add `import React from 'react'` if not present.

This exposes cache metadata to consuming components without changing the existing query behavior.
