describe('setWithJitter', () => {
  it('should apply jitter to TTL within 10% variance (D-07, D-08, D-09)', async () => {
    // Mock Math.random to test jitter bounds
    const randomSpy = vi.spyOn(Math, 'random');

    // Test lower bound: random = 0 -> multiplier = 0.9
    randomSpy.mockReturnValue(0);
    const baseTtl = 300;
    const expectedLower = Math.floor(baseTtl * 0.9); // 270

    // We can't directly test the TTL passed to Redis without mocking deeper,
    // but we can verify the method doesn't throw and returns true
    const result1 = await cacheService.setWithJitter('test:jitter:low', { test: 'value' }, baseTtl);
    // Result depends on Redis availability; if unavailable, returns false (graceful degradation)
    expect(typeof result1).toBe('boolean');

    // Test upper bound: random = 1 -> multiplier = 1.1
    randomSpy.mockReturnValue(1);
    const expectedUpper = Math.floor(baseTtl * 1.1); // 330

    const result2 = await cacheService.setWithJitter('test:jitter:high', { test: 'value' }, baseTtl);
    expect(typeof result2).toBe('boolean');

    randomSpy.mockRestore();
  });

  it('should use CACHE_TTL.MEDIUM as default when no TTL provided', async () => {
    const result = await cacheService.setWithJitter('test:default-ttl', { data: 123 });
    expect(typeof result).toBe('boolean');
  });

  it('should return false when Redis unavailable (graceful degradation)', async () => {
    // Mock isAvailable to return false
    const isAvailableSpy = vi.spyOn(cacheService, 'isAvailable').mockReturnValue(false);

    const result = await cacheService.setWithJitter('test:unavailable', { data: 'test' }, 300);
    expect(result).toBe(false);

    isAvailableSpy.mockRestore();
  });

  it('should handle errors gracefully and return false', async () => {
    // Mock isAvailable to return true but client to throw
    const isAvailableSpy = vi.spyOn(cacheService, 'isAvailable').mockReturnValue(true);

    // Access private client via any cast to mock setex
    const client = (cacheService as unknown as { client: { setex: () => Promise<void> } }).client;
    if (client) {
      const setexSpy = vi.spyOn(client, 'setex' as keyof typeof client).mockRejectedValue(new Error('Redis error'));

      const result = await cacheService.setWithJitter('test:error', { data: 'test' }, 300);
      expect(result).toBe(false);

      setexSpy.mockRestore();
    }

    isAvailableSpy.mockRestore();
  });
});

/**
 * ETag Middleware Tests (Phase 33)
 */

import { describe, it, expect, vi, beforeEach } from 'vitest';
import type { Request, Response, NextFunction } from 'express';
import { etagMiddleware } from './etagMiddleware';

// Helper to create mock request
const mockRequest = (overrides: Partial<Request> = {}): Request => ({
  method: 'GET',
  path: '/api/news',
  get: vi.fn().mockReturnValue(undefined),
  ...overrides,
} as unknown as Request);

// Helper to create mock response
const mockResponse = (): Response & { _jsonBody?: unknown } => {
  const res = {
    json: vi.fn(function (this: Response & { _jsonBody?: unknown }, body: unknown) {
      this._jsonBody = body;
      return this;
    }),
    set: vi.fn().mockReturnThis(),
    status: vi.fn().mockReturnThis(),
    end: vi.fn().mockReturnThis(),
  };
  return res as unknown as Response & { _jsonBody?: unknown };
};

describe('etagMiddleware', () => {
  let req: Request;
  let res: Response & { _jsonBody?: unknown };
  let next: NextFunction;

  beforeEach(() => {
    req = mockRequest();
    res = mockResponse();
    next = vi.fn();
  });

  describe('GET requests', () => {
    it('should generate weak ETag header for JSON responses (D-04, D-05)', () => {
      etagMiddleware(req, res, next);
      expect(next).toHaveBeenCalled();

      // Call the overridden json method
      const testBody = { articles: [{ id: '1', title: 'Test' }] };
      res.json(testBody);

      // Verify ETag was set
      expect(res.set).toHaveBeenCalledWith('ETag', expect.stringMatching(/^W\/"[a-f0-9]{16}"$/));
    });

    it('should return 304 when If-None-Match matches ETag (D-06)', () => {
      // First, determine the ETag for our test body
      const testBody = { data: 'test' };

      // Mock request with If-None-Match header
      // We need to calculate the expected ETag first
      const crypto = require('crypto');
      const content = JSON.stringify(testBody);
      const hash = crypto.createHash('md5').update(content).digest('hex').slice(0, 16);
      const expectedEtag = `W/"${hash}"`;

      req = mockRequest({
        method: 'GET',
        get: vi.fn((header: string) => {
          if (header === 'If-None-Match') return expectedEtag;
          return undefined;
        }),
      });

      etagMiddleware(req, res, next);
      expect(next).toHaveBeenCalled();

      // Call json with same body
      res.json(testBody);

      // Should return 304, not the body
      expect(res.status).toHaveBeenCalledWith(304);
      expect(res.end).toHaveBeenCalled();
    });

    it('should return full response when If-None-Match does not match', () => {
      req = mockRequest({
        method: 'GET',
        get: vi.fn((header: string) => {
          if (header === 'If-None-Match') return 'W/"stale-etag-12345"';
          return undefined;
        }),
      });

      etagMiddleware(req, res, next);
      expect(next).toHaveBeenCalled();

      const testBody = { fresh: 'data' };
      res.json(testBody);

      // Should set new ETag and return body
      expect(res.set).toHaveBeenCalledWith('ETag', expect.stringMatching(/^W\/"[a-f0-9]{16}"$/));
      expect(res._jsonBody).toEqual(testBody);
    });
  });

  describe('non-GET requests', () => {
    it('should skip POST requests', () => {
      req = mockRequest({ method: 'POST' });

      etagMiddleware(req, res, next);
      expect(next).toHaveBeenCalled();

      // json should not be overridden
      const originalJson = res.json;
      res.json({ data: 'post' });
      expect(res.set).not.toHaveBeenCalledWith('ETag', expect.any(String));
    });

    it('should skip PUT requests', () => {
      req = mockRequest({ method: 'PUT' });

      etagMiddleware(req, res, next);
      expect(next).toHaveBeenCalled();
    });

    it('should skip DELETE requests', () => {
      req = mockRequest({ method: 'DELETE' });

      etagMiddleware(req, res, next);
      expect(next).toHaveBeenCalled();
    });
  });

  describe('skip paths', () => {
    it('should skip /health endpoint', () => {
      req = mockRequest({ path: '/health' });

      etagMiddleware(req, res, next);
      expect(next).toHaveBeenCalled();

      // Verify json is not overridden
      res.json({ status: 'healthy' });
      expect(res.set).not.toHaveBeenCalledWith('ETag', expect.any(String));
    });

    it('should skip /readiness endpoint', () => {
      req = mockRequest({ path: '/readiness' });

      etagMiddleware(req, res, next);
      expect(next).toHaveBeenCalled();
    });

    it('should skip /metrics endpoint', () => {
      req = mockRequest({ path: '/metrics' });

      etagMiddleware(req, res, next);
      expect(next).toHaveBeenCalled();
    });

    it('should skip /api/health endpoints', () => {
      req = mockRequest({ path: '/api/health/db' });

      etagMiddleware(req, res, next);
      expect(next).toHaveBeenCalled();
    });
  });

  describe('ETag format', () => {
    it('should generate consistent ETag for same content', () => {
      const testBody = { id: 1, name: 'test' };

      // First request
      etagMiddleware(req, res, next);
      res.json(testBody);
      const firstEtag = (res.set as ReturnType<typeof vi.fn>).mock.calls.find(
        (call) => call[0] === 'ETag'
      )?.[1];

      // Reset and second request with same body
      res = mockResponse();
      req = mockRequest();
      next = vi.fn();

      etagMiddleware(req, res, next);
      res.json(testBody);
      const secondEtag = (res.set as ReturnType<typeof vi.fn>).mock.calls.find(
        (call) => call[0] === 'ETag'
      )?.[1];

      expect(firstEtag).toBe(secondEtag);
    });

    it('should generate different ETag for different content', () => {
      // First request
      etagMiddleware(req, res, next);
      res.json({ content: 'A' });
      const firstEtag = (res.set as ReturnType<typeof vi.fn>).mock.calls.find(
        (call) => call[0] === 'ETag'
      )?.[1];

      // Reset and second request with different body
      res = mockResponse();
      req = mockRequest();
      next = vi.fn();

      etagMiddleware(req, res, next);
      res.json({ content: 'B' });
      const secondEtag = (res.set as ReturnType<typeof vi.fn>).mock.calls.find(
        (call) => call[0] === 'ETag'
      )?.[1];

      expect(firstEtag).not.toBe(secondEtag);
    });

    it('should use weak validator format W/"..."', () => {
      etagMiddleware(req, res, next);
      res.json({ test: true });

      const etagCall = (res.set as ReturnType<typeof vi.fn>).mock.calls.find(
        (call) => call[0] === 'ETag'
      );
      expect(etagCall?.[1]).toMatch(/^W\/"[a-f0-9]+"/);
    });
  });
});
