Worker mode (RUN_JOBS=true) constructs NewsAggregator + CleanupService + email-digest schedulers AND initializes the Socket.IO Emitter so cross-replica broadcasts work.

Without this plan, scaling to N replicas breaks immediately: replica A's NewsAggregator Map has different data than replica B's, and any of N replicas could spawn its own RSS aggregation loop.

    export async function getArticles(opts: ListOptions = {}): Promise<{ articles: NewsArticle[]; total: number }> {
      const cache = CacheService.getInstance();
      const cacheKey = CacheKeys.newsList(JSON.stringify(opts));
      return cache.getOrSet(cacheKey, async () => {
        const where: Record<string, unknown> = {};
        if (opts.regions?.length) where.perspective = { in: opts.regions };
        if (opts.sentiment) where.sentiment = opts.sentiment;
        if (opts.search) {
          where.OR = [
            { title: { contains: opts.search, mode: 'insensitive' } },
            { content: { contains: opts.search, mode: 'insensitive' } },
          ];
        }
        // topics filter — JSONB string-contains, preserves existing schema (apps/web/prisma/schema.prisma:46 GIN index)
        if (opts.topics?.length) {
          where.AND = opts.topics.map(t => ({ topics: { contains: t } }));
        }
        const [articles, total] = await prisma.$transaction([
          prisma.newsArticle.findMany({
            where,
            take: opts.limit ?? 20,
            skip: opts.offset ?? 0,
            orderBy: { publishedAt: 'desc' },
            include: { source: true },
          }),
          prisma.newsArticle.count({ where }),
        ]);
        return { articles: articles.map(mapToNewsArticle), total };
      }, CACHE_TTL.SHORT);
    }
