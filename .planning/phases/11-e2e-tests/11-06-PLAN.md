import { test, expect } from '@playwright/test';

// These tests run with storageState from auth.setup.ts
// User is already authenticated when tests start

test.describe('Bookmarks Page (Authenticated)', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/bookmarks');
    await page.waitForLoadState('networkidle');
  });

  test('should load the Bookmarks page', async ({ page }) => {
    await expect(page).toHaveURL('/bookmarks');

    // Check for page header
    const heading = page.locator('h1:has-text("Gespeicherte Artikel")');
    await expect(heading).toBeVisible();
  });

  test('should display page description', async ({ page }) => {
    // Description text about bookmarks
    const description = page.locator('text=/gespeicherten|saved|bookmarks/i').first();
    await expect(description).toBeVisible();
  });

  test('should show empty state or articles', async ({ page }) => {
    // Either empty state OR articles grid should be visible
    const emptyText = page.locator('text=Keine gespeicherten Artikel');
    const articleGrid = page.locator('.grid.gap-4');

    const hasEmpty = await emptyText.isVisible();
    const hasArticles = await articleGrid.isVisible();

    // One of these must be true
    expect(hasEmpty || hasArticles).toBeTruthy();
  });

  test('should display empty state icon when no bookmarks', async ({ page }) => {
    const emptyText = page.locator('text=Keine gespeicherten Artikel');

    if (await emptyText.isVisible()) {
      // Empty state should have bookmark icon
      const emptyIcon = page.locator('svg.h-16.w-16, [class*="lucide-bookmark"]');
      await expect(emptyIcon.first()).toBeVisible();
    }
  });

  test('should display clear all button when bookmarks exist', async ({ page }) => {
    const articleGrid = page.locator('.grid.gap-4');

    if (await articleGrid.isVisible()) {
      // "Alle entfernen" button should be visible
      const clearAllBtn = page.locator('button:has-text("Alle entfernen")');
      await expect(clearAllBtn).toBeVisible();
    }
  });

  test('should display bookmark count', async ({ page }) => {
    const articleGrid = page.locator('.grid.gap-4');

    if (await articleGrid.isVisible()) {
      // Count indicator (e.g., "X Artikel gespeichert")
      const countText = page.locator('text=/\\d+ Artikel/i');
      if (await countText.isVisible()) {
        await expect(countText).toBeVisible();
      }
    }
  });

  test('should have glass panel styling', async ({ page }) => {
    // Page uses glass-panel for article cards
    const glassPanels = page.locator('.glass-panel, [class*="rounded-lg"]');
    const panelCount = await glassPanels.count();
    expect(panelCount).toBeGreaterThanOrEqual(0); // May be 0 if empty
  });

  test('should navigate from Profile quick action', async ({ page }) => {
    // Go to profile first
    await page.goto('/profile');
    await page.waitForLoadState('networkidle');

    // Click Bookmarks quick action
    const bookmarksBtn = page.locator('button:has-text("Bookmarks")');
    await bookmarksBtn.click();

    await expect(page).toHaveURL('/bookmarks');
  });
});
