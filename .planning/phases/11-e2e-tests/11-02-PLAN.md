import { test, expect } from '@playwright/test';

test.describe('Dashboard Page', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/');
    await page.waitForLoadState('networkidle');
  });

  test('should load the Dashboard page', async ({ page }) => {
    await expect(page).toHaveURL('/');
    // Check for search input
    const searchInput = page.locator('input[placeholder*="Search signals"]');
    await expect(searchInput).toBeVisible();
  });

  test('should display news feed with articles', async ({ page }) => {
    // Wait for articles to load - SignalCard has glass-panel styling
    const articleCard = page.locator('.glass-panel').first();
    await expect(articleCard).toBeVisible({ timeout: 10000 });
  });

  test('should have view toggle buttons', async ({ page }) => {
    // Grid button (LayoutGrid icon)
    const gridButton = page.locator('button').filter({ has: page.locator('[class*="lucide-layout-grid"]') });
    // List button (List icon)
    const listButton = page.locator('button').filter({ has: page.locator('[class*="lucide-list"]') });

    // At least one should be visible
    const gridVisible = await gridButton.count() > 0;
    const listVisible = await listButton.count() > 0;
    expect(gridVisible || listVisible).toBeTruthy();
  });

  test('should toggle between grid and list views', async ({ page }) => {
    // Find view toggle area
    const viewToggle = page.locator('.flex.gap-1, .flex.gap-2').filter({ has: page.locator('button') }).first();

    if (await viewToggle.isVisible()) {
      // Get the buttons
      const buttons = viewToggle.locator('button');
      const buttonCount = await buttons.count();

      if (buttonCount >= 2) {
        // Click second button (list view)
        await buttons.nth(1).click();
        await page.waitForTimeout(300); // Animation

        // Grid container should change
        const gridContainer = page.locator('.grid');
        await expect(gridContainer.first()).toBeVisible();
      }
    }
  });

  test('should display trend filter buttons', async ({ page }) => {
    // Escalation filter
    const escalationBtn = page.locator('button:has-text("Escalation")');
    // De-escalation filter
    const deEscalationBtn = page.locator('button:has-text("De-escalation")');

    // At least escalation should be visible
    await expect(escalationBtn).toBeVisible();
  });

  test('should filter by escalation trend', async ({ page }) => {
    const escalationBtn = page.locator('button:has-text("Escalation")');

    if (await escalationBtn.isVisible()) {
      await escalationBtn.click();
      await page.waitForTimeout(300);

      // Button should show active state (red border for escalation)
      await expect(escalationBtn).toHaveClass(/border-\[#ff0044\]|border-red/);
    }
  });

  test('should filter by de-escalation trend', async ({ page }) => {
    const deEscalationBtn = page.locator('button:has-text("De-escalation")');

    if (await deEscalationBtn.isVisible()) {
      await deEscalationBtn.click();
      await page.waitForTimeout(300);

      // Button should show active state (green border for de-escalation)
      await expect(deEscalationBtn).toHaveClass(/border-\[#00ff88\]|border-green/);
    }
  });

  test('should display cache indicator', async ({ page }) => {
    // CacheIndicator shows cache status
    const cacheIndicator = page.locator('text=/cache|cached|fresh/i');
    // May or may not be visible depending on cache state
  });

  test('should navigate from sidebar', async ({ page }) => {
    // Navigate away first
    await page.goto('/monitor');
    await page.waitForLoadState('networkidle');

    // Click home/dashboard link
    const homeLink = page.locator('a[href="/"]').first();
    await homeLink.click();

    await expect(page).toHaveURL('/');
  });

  test('should display AskAI component', async ({ page }) => {
    // AskAI has a distinctive input or button
    const askAI = page.locator('text=/ask|fragen/i').first();
    // Component may or may not be visible depending on data
    if (await askAI.isVisible()) {
      await expect(askAI).toBeVisible();
    }
  });
});
