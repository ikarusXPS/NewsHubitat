```typescript
import { test, expect } from '@playwright/test';

// These tests run with storageState from auth.setup.ts
// User is already authenticated when tests start

test.describe('Settings Page (Authenticated)', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/settings');
    await page.waitForLoadState('networkidle');
  });

  test('should load the Settings page', async ({ page }) => {
    await expect(page).toHaveURL('/settings');

    // Check for Settings header (may be in sidebar or page)
    const heading = page.locator('h1, h2').filter({ hasText: /Settings|Einstellungen/i }).first();
    await expect(heading).toBeVisible();
  });

  test('should display theme toggle buttons', async ({ page }) => {
    // Dark Mode button
    const darkModeBtn = page.locator('button:has-text("Dark Mode")').first();
    await expect(darkModeBtn).toBeVisible();

    // Light Mode button
    const lightModeBtn = page.locator('button:has-text("Light Mode")').first();
    await expect(lightModeBtn).toBeVisible();
  });

  test('should toggle to light theme', async ({ page }) => {
    const lightModeBtn = page.locator('button:has-text("Light Mode")').first();
    await lightModeBtn.click();
    await page.waitForTimeout(300);

    // Button should show active state (border color)
    await expect(lightModeBtn).toHaveClass(/border-blue-500|border-\[#00f0ff\]/);
  });

  test('should toggle to dark theme', async ({ page }) => {
    // First switch to light, then back to dark
    const lightModeBtn = page.locator('button:has-text("Light Mode")').first();
    const darkModeBtn = page.locator('button:has-text("Dark Mode")').first();

    await lightModeBtn.click();
    await page.waitForTimeout(300);

    await darkModeBtn.click();
    await page.waitForTimeout(300);

    // Dark mode button should be active
    await expect(darkModeBtn).toHaveClass(/border-blue-500|border-\[#00f0ff\]/);
  });

  test('should display language toggle buttons', async ({ page }) => {
    // Deutsch button
    const deutschBtn = page.locator('button:has-text("Deutsch")');
    await expect(deutschBtn).toBeVisible();

    // English button
    const englishBtn = page.locator('button:has-text("English")');
    await expect(englishBtn).toBeVisible();
  });

  test('should toggle language to English', async ({ page }) => {
    const englishBtn = page.locator('button:has-text("English")');
    await englishBtn.click();
    await page.waitForTimeout(300);

    // Button should show active state
    await expect(englishBtn).toHaveClass(/border-blue-500|border-\[#00f0ff\]/);
  });

  test('should toggle language to Deutsch', async ({ page }) => {
    const deutschBtn = page.locator('button:has-text("Deutsch")');
    await deutschBtn.click();
    await page.waitForTimeout(300);

    // Button should show active state
    await expect(deutschBtn).toHaveClass(/border-blue-500|border-\[#00f0ff\]/);
  });

  test('should display export settings button', async ({ page }) => {
    // Export button
    const exportBtn = page.locator('button:has-text("Einstellungen exportieren"), button:has-text("Export")');
    await expect(exportBtn.first()).toBeVisible();
  });

  test('should display import settings button', async ({ page }) => {
    // Import button
    const importBtn = page.locator('button:has-text("Einstellungen importieren"), button:has-text("Import")');
    await expect(importBtn.first()).toBeVisible();
  });

  test('should display region selection section', async ({ page }) => {
    // Region selection has region buttons or checkboxes
    const regionSection = page.locator('text=/Region|Regionen/i').first();
    if (await regionSection.isVisible()) {
      await expect(regionSection).toBeVisible();
    }
  });

  test('should have glass panel styling', async ({ page }) => {
    // Settings page uses glass-panel for sections
    const glassPanels = page.locator('.glass-panel');
    const panelCount = await glassPanels.count();
    expect(panelCount).toBeGreaterThan(0);
  });

  test('should navigate from Profile quick action', async ({ page }) => {
    // Go to profile first
    await page.goto('/profile');
    await page.waitForLoadState('networkidle');

    // Click Settings quick action
    const settingsBtn = page.locator('button:has-text("Settings")');
    await settingsBtn.click();

    await expect(page).toHaveURL('/settings');
  });
});
```
