import { test, expect } from '@playwright/test';

// These tests run with storageState from auth.setup.ts
// User is already authenticated when tests start

test.describe('Reading History Page (Authenticated)', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/history');
    await page.waitForLoadState('networkidle');
  });

  test('should load the Reading History page', async ({ page }) => {
    await expect(page).toHaveURL('/history');

    // Check for page header (Clock icon and title)
    const heading = page.locator('h1, h2').filter({ hasText: /Reading History|Leseverlauf/i }).first();
    await expect(heading).toBeVisible();
  });

  test('should display history stats section', async ({ page }) => {
    // HistoryStats component shows reading statistics
    const statsSection = page.locator('[class*="stat"], text=/articles|read/i').first();
    if (await statsSection.isVisible()) {
      await expect(statsSection).toBeVisible();
    }
  });

  test('should show empty state or timeline groups', async ({ page }) => {
    // Either empty state OR timeline groups should be visible
    const emptyText = page.locator('text=/No Reading History|Kein Leseverlauf/i');
    const timelineGroup = page.locator('h2').filter({ hasText: /Today|Yesterday|This Week|Older|Heute|Gestern/i }).first();

    const hasEmpty = await emptyText.isVisible();
    const hasTimeline = await timelineGroup.isVisible();

    // One of these must be true
    expect(hasEmpty || hasTimeline).toBeTruthy();
  });

  test('should display timeline group headers when history exists', async ({ page }) => {
    const timelineGroup = page.locator('h2').filter({ hasText: /Today|Yesterday|This Week|Older|Heute|Gestern/i }).first();

    if (await timelineGroup.isVisible()) {
      await expect(timelineGroup).toBeVisible();
    }
  });

  test('should display filter buttons', async ({ page }) => {
    // HistoryFilters has date preset buttons
    const todayFilter = page.locator('button:has-text("Today"), button:has-text("Heute")');
    const weekFilter = page.locator('button:has-text("This Week"), button:has-text("Diese Woche")');

    // At least one filter should be visible
    const hasTodayFilter = await todayFilter.count() > 0;
    const hasWeekFilter = await weekFilter.count() > 0;

    expect(hasTodayFilter || hasWeekFilter).toBeTruthy();
  });

  test('should filter by Today preset', async ({ page }) => {
    const todayFilter = page.locator('button:has-text("Today"), button:has-text("Heute")').first();

    if (await todayFilter.isVisible()) {
      await todayFilter.click();
      await page.waitForTimeout(300);

      // Filter button should show active state
      await expect(todayFilter).toHaveClass(/active|bg-|border-/);
    }
  });

  test('should have clear history button', async ({ page }) => {
    // Clear history button
    const clearBtn = page.locator('button:has-text("Clear History"), button:has-text("Verlauf loschen"), button:has-text("Clear")');
    await expect(clearBtn.first()).toBeVisible();
  });

  test('should display empty state icon when no history', async ({ page }) => {
    const emptyText = page.locator('text=/No Reading History|Kein Leseverlauf/i');

    if (await emptyText.isVisible()) {
      // Empty state should have icon (Clock or similar)
      const emptyIcon = page.locator('svg.h-16, [class*="lucide-clock"]').first();
      if (await emptyIcon.isVisible()) {
        await expect(emptyIcon).toBeVisible();
      }
    }
  });

  test('should have glass panel styling', async ({ page }) => {
    // Page uses glass-panel for sections
    const glassPanels = page.locator('.glass-panel');
    const panelCount = await glassPanels.count();
    expect(panelCount).toBeGreaterThanOrEqual(0);
  });

  test('should navigate from Profile quick action', async ({ page }) => {
    // Go to profile first
    await page.goto('/profile');
    await page.waitForLoadState('networkidle');

    // Click History quick action
    const historyBtn = page.locator('button:has-text("History")');
    await historyBtn.click();

    await expect(page).toHaveURL('/history');
  });
});
